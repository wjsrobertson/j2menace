005    package com.rattat.micro.game.frog.board;
006    
007    /**
008     * Very simple class - just holds public static final ints that 
009     * are used to notify BoardListeners of events. In most Observer pattern
010     * implementations an instance of the Event class would be passed around
011     * not notify of events, but creating all these objects is not a great thing in 
012     * MIDP. Passing around ints insatead is plenty effective. 
013     *
014     * @author william@rattat.com
015     */
016    public class BoardEvent {
017        
018        /**
019         * User has lost a life
020         */
021        public static final int FROG_DEATH = 1;
022        
023        /**
024         * User has completed a level
025         */
026        public static final int LEVEL_COMPLETE = 2;
027        
028        /**
029         * User has won the game
030         */
031        public static final int GAME_WON = 3;
032        
033        /**
034         * User has lost the game
035         */
036        public static final int GAME_LOST = 4;
037        
038        /**
039         * User has started a new level
040         */
041        public static final int LEVEL_START = 5;
042        
043        /**
044         * User has started a new game
045         */
046        public static final int GAME_START = 6;
047        
048        /**
049         * User has moved the frog
050         */
051        public static final int FROG_MOVE = 7;
052        
053        /**
054         * User has managed to get the frog into one of the home spaces
055         * at the top of the screen
056         */
057        public static final int FROG_HOME = 8;
058        
059        /** 
060         * Private constructor - this class should never be instanciated. Just 
061         * used to get at the static ints
062         */
063        private BoardEvent () {
064        }
065        
066    }
