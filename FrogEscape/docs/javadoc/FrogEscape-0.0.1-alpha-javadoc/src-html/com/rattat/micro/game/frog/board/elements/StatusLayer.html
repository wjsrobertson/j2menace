008    package com.rattat.micro.game.frog.board.elements;
009    
010    import com.rattat.micro.graphics.Layer;
011    import javax.microedition.lcdui.Graphics;
012    import javax.microedition.lcdui.Image;
013    
014    /**
015     *
016     * @author william@rattat.com
017     */
018    public class StatusLayer extends Layer {
019    
020        /**
021         * Number of lives to display
022         */
023        private int lives = 1;
024        
025        /**
026         * The amount of time a player has to complete a level
027         */
028        private int maxTime = 100;
029        
030        /**
031         * The remaining time a player has to complete the level
032         */
033        private int time = 0;
034        
035        /**
036         * The current level that the player is on
037         */
038        private int level = 1;
039        
040        /**
041         * The width of one cell of this layer. Used for rendering
042         */
043        int cellWidth = 16;
044        
045        /**
046         * When the time reaches this point it will be displayed to indicate urgency
047         */
048        private static final int CTRITICAL_TIME_PERCENT = 0;
049            
050        /**
051         * Width of this component in cells
052         */
053        private static final int NUM_CELLS_WIDE = 15;
054        
055        /**
056         * Path to the image file saying "Time"
057         */
058        private String timeImageFile = "/__time.png";
059        
060        /**
061         * Image saying "Time"
062         */
063        private Image timeImage = null;
064        
065        /**
066         * Path to the image representing one player life
067         */
068        private String lifeImageFile = "/frog_life.png";
069        
070        /**
071         * Image representing one player life
072         */
073        private Image lifeImage = null;
074        
075        /**
076         * Path to the image representing one level
077         */
078        private String levelImageFile = "/level.png";
079        
080        /**
081         * Image representing one level
082         */
083        private Image levelImage = null;
084        
085        /**
086         * X offset to the time image
087         */
088        private int timeImageXOffset;
089        
090        /**
091         * X offset from the RHS of the board where the first level image should
092         * appear
093         */
094        private int levelImageXOffset;  // offset from the RHS
095    
096        /**
097         * The width of the life bar when it is full
098         */
099        private int timeBarFullWidth = 0;
100        
101        /**
102         * The normal colour of the time bar
103         */
104        private int timeBarFullColor = 0x00FF00;
105        
106        /**
107         * Calculated from timeBarFullWidth - if the bar is any smaller
108         * than this then it is critical
109         */
110        private int timeBarCriticalWidth = 200;
111        
112        /**
113         * The colour that the life bar will be rendered in when time is critical
114         */
115        private int timeBarCriticalColor = 0xFF0000;
116        
117        /**
118         * y padding in pixels 
119         */
120        private static final  int LIFE_BAR_PADDING_Y = 1;
121        
122        /**
123         * X padding in pixels 
124         */
125        private static final int LIFE_BAR_PADDING_X = 1;
126    
127        /** 
128         * Creates a new instance of ScoreLayer 
129         */
130        public StatusLayer() {
131            init();
132        }
133    
134        /**
135         * Load all images and get the StatusLayer ready for use
136         */
137        private void init() {
138            try {
139                timeImage = Image.createImage(timeImageFile);
140                timeImageXOffset = getWidth() - timeImage.getWidth();
141            } catch (Exception e) {
142                System.err.println("Error loading image: " + e);
143            }
144            
145            try {
146                lifeImage = Image.createImage(lifeImageFile);
147            } catch (Exception e) {
148                System.err.println("Error loading image: " + e);
149            }
150            
151            try {
152                levelImage = Image.createImage(levelImageFile);
153                levelImageXOffset = 2 * levelImage.getWidth();
154            } catch (Exception e) {
155                System.err.println("Error loading image: " + e);
156            }
157                    
158            setWidth( cellWidth * NUM_CELLS_WIDE );
159            setHeight( timeImage.getHeight() );
160            
161            timeBarFullWidth    = getWidth() - (2 * timeImage.getWidth());
162            timeBarCriticalWidth = (int) ((double)timeBarFullWidth / 10);
163        }
164        
165        /**
166         * Reset back to initial state
167         */
168        private void reset() {
169            level = 1;
170            time = maxTime;
171        }
172    
173        /**
174         * Set the amount of time a user is allowed to play for before they loose
175         * a life
176         */
177        public void setMaxTime(int maxTime) {
178            this.maxTime = maxTime;
179        }
180    
181        /**
182         * Set the remaining time
183         *
184         * @param int time
185         */
186        public void setTime(int time) {
187            this.time = time;
188        }
189        
190        /**
191         * Get the remaining time
192         */
193        public int getTime() {
194            return time;
195        }
196        
197        /**
198         * Set the number of lives a player has left
199         *
200         * @param int lives
201         */
202        public void setLives (int lives) {
203            this.lives = lives;
204        }
205    
206        /**
207         * Set the current level
208         *
209         * @param int level
210         */
211        public void setLevel (int level) {
212            this.level = level;
213        }
214        
215        /**
216         * Paint the StatueLayer on a Graphics object
217         *
218         * @param Graphics g
219         */
220        public void paint(Graphics g) {
221            
222            // clear the rectangle
223            
224            g.setColor(0, 0, 0); // black
225            g.fillRect(getX(), getY(), getWidth(), getHeight());
226            
227            // paint the time image (RHS of time bar)
228            
229            if (timeImage != null) {
230                g.drawImage(timeImage, getWidth() - timeImage.getWidth(), getY(), Graphics.TOP|Graphics.LEFT);
231            }
232    
233            // draw the lives
234            
235            if (lifeImage != null ) {
236                for (int i=0 ; i<lives-1 ; i++) {
237                    g.drawImage(lifeImage, i * lifeImage.getWidth(), getY() , Graphics.TOP|Graphics.LEFT);
238                }
239            }
240            
241            // paint the level indicators
242            if (levelImage != null) {
243                for (int i=0 ; i<level ; i++) {
244                    int xOffset = getWidth() - levelImageXOffset - (i * lifeImage.getWidth());
245                    g.drawImage(levelImage, xOffset, getY() , Graphics.TOP|Graphics.LEFT);
246                }
247            }
248            
249            // paint the time strip
250    
251            if (timeImage != null && time > 0) {
252                int color = timeBarFullColor;
253                
254                if ( (((double)time / (double)maxTime) * 100) <= CTRITICAL_TIME_PERCENT ) {
255                    color = timeBarCriticalColor;
256                }
257                
258                g.setColor(color);
259                
260                int barWidth  = (int) ((((double)time) / ((double) maxTime)) * timeBarFullWidth);
261                int barEndX   = getX() + getWidth() - timeImage.getWidth() - LIFE_BAR_PADDING_X;
262                int barStartX = barEndX - ((int) barWidth) - LIFE_BAR_PADDING_X;
263                
264                int barHeight = timeImage.getHeight() / 2;
265                int barEndY   = getY() + getHeight() - LIFE_BAR_PADDING_Y;
266                int barStartY = barEndY - barHeight - LIFE_BAR_PADDING_Y;
267                
268                g.fillRect( barStartX, barStartY, barWidth, barHeight );
269            }
270        }    
271        
272        /**
273         * Decrement the time counter by one unit
274         */
275        public void tick() {
276            time--;
277        }
278    }
