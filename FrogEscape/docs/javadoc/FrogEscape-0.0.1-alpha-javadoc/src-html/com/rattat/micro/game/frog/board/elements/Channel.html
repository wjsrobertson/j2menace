219        /**
220         * Create a channel instance
221         * 
222         * cellContents paramater is an array containing which cells the Channel 
223         * consists of. e.g. {CELL_BG_ROAD, BGROAD, CELL_CAR1, CELL_BG_ROAD} would be a four cell
224         * wide channel.
225         * 
226         * 
227         * 
228         * @param int[] cellContents
229         * @param int velocity
230         * @return Channel
231         * @throws Exception
232         */
233        public static Channel create(int[] cellContents, int velocity) throws Exception {
234            Image tileImage = getImage();
235    
236            int cellHeight = tileImage.getHeight();
237            int cellWidth  = tileImage.getHeight(); // yes, width = cellHeight (square)
238    
239            Channel channel = new Channel(cellContents.length, tileImage, cellWidth, cellHeight);
240    
241            channel.setCellContents( cellContents );
242    
243            channel.setVelocity( velocity );
244            
245            int direction = (velocity > 0) ? RIGHT : LEFT;
246            channel.setDirection( direction );
247            
248            //int scrollBoundary = (velocity < 0) ? channel.getWidth() + channel.getCellWidth () : -channel.getCellWidth();
249            int scrollBoundary = -channel.getCellWidth();
250            channel.setScrollBoundary( scrollBoundary );
251    
252            channel.init();
253            
254            return channel;
255        }

282         * Initialise the channel - Setup the cell contents according to the 
283         * cellContents array

285        private void init() {
286            for (int i=0 ; i<cellContents.length ; i++) {
287                setCell(i, cellContents[i]);
288            }
289        }

291        /**
292         * Update the channel 
293         *
294         * move acording to velocity then update position & cell contents to give 
295         * the impression on a constantly moving image
296         */
297        public void update() {
298            // move according to velocity & update cell contents according to position
299            
300            if (velocity != 0) {
301                move(velocity, 0);
302                int offsetPixels = 0;
303                
304                if (direction == LEFT) {
305                    if ( (getX() + getCellWidth())  < scrollBoundary) {
306                        offsetPixels = scrollBoundary - getCellWidth() - getX();
307                    }
308                } else if (direction == RIGHT) {
309                    if ( getX() > scrollBoundary) {
310                        offsetPixels = scrollBoundary -  getX();
311                    }
312                }
313                
314                int offsetCells = offsetPixels / getCellWidth();
315                if (offsetCells != 0) {
316                    shuntCells(offsetCells);
317                    moveByCells(offsetCells); 
318                }
319           } 
320        }
321        
322        /**
323         * Shift the TiledLayer cell contents by a pixel offset
324         *
325         * @param int offset
326         */
327        private void shuntCells(int offset) {
328            currentOffset += offset;
329            
330            // modulo arithmetic in base cellContents.length
331            // transform the currentOffset to a positive value less than cellContents.length
332            
333            if (currentOffset < 0) {
334                currentOffset += cellContents.length;
335            }
336            currentOffset = currentOffset % cellContents.length;
337    
338            for (int i=0 ; i<cellContents.length ; i++) {
339                int index = ( i +currentOffset ) % cellContents.length;
340                setCell(i, cellContents[index]);
341            }
342        }

346         *
347         * @param int numCells
348         */
349        public void moveByCells(int numCells) {
350            int pixelMoveX = numCells * getCellWidth();
351            
352            move(pixelMoveX, 0);
353        }

364        /**
365         * Check if Sprite is colliding with a specific cell
366         *
367         * @param Sprite sprite
368         * @param int cell
369         */
370        private boolean isColliding(Sprite sprite, int cell) {
371            
372            /*
373             allow some leeway for the user in the form of padding inside
374             the sprites borders
375            */
376            int padding = (int) ((double)sprite.getWidth() / 4);
377            
378            if (padding == 0) {
379                padding = 1;
380            }
381            
382            int spriteMinX = sprite.getX() + padding;
383            int spriteMaxX = sprite.getX() + sprite.getWidth() - padding;
384            int spriteMinY = sprite.getY() + padding;
385            int spriteMaxY = sprite.getY() + sprite.getHeight() - padding;
386            
387            int cellMinY = getY();
388            int cellMaxY = getY() + getCellHeight();
389            
390            // check for y collision
391            
392            if ( spriteMinY > cellMinY && spriteMinY < cellMaxY 
393                 || spriteMaxY > cellMinY && spriteMaxY < cellMaxY) {
394                
395            } else {
396                return false;
397            }
398            
399            // check for X collision
400            
401            int cellMinX = getX() + (cell) * getCellWidth();
402            int cellMaxX = getX() + (cell+1) * getCellWidth();
403            
404            if (  (spriteMinX > cellMinX && spriteMinX < cellMaxX)
405                   || (spriteMaxX > cellMinX && spriteMaxX < cellMaxX)
406                   ) {
407    
408                   return true;
409            }
410            
411            return false;
412        }

414        /**
415         * Check if a sprite is colliding with the Channel, and if so, is
416         * the cell it is colliding with a deadly cell
417         *
418         * @param Sprite sprite
419         */
420        public boolean isDeadlyCollision(Sprite sprite) {      
421    
422            for (int i=0 ; i<getColumns() ; i++) {
423                if (isColliding(sprite, i)) {
424                    if ( cellType(i) == TYPE_DEADLY ) {
425                        return true;
426                    }
427                }
428            }
429    
430            return false;
431        }

433        /**
434         * Check is a sprite is landing on the Channel.
435         *
436         * We consider the Sprite to be landing on the 
437         * Channel if its centre pixel is within the bounds
438         * of the layer.
439         *
440         * @param Sprite sprite
441         */
442        public boolean isLanding(Sprite sprite) {
443            int spriceCentreX = sprite.getX() + (sprite.getWidth()/2);
444            int spriteCentreY = sprite.getY() + (sprite.getHeight()/2);
445            
446            return     spriceCentreX >= getX()
447                    && spriteCentreY >= getY()
448                    && spriceCentreX < (getX() + getWidth()) 
449                    && spriteCentreY < (getY() + getHeight());
450        }
451        
452        /**
453         * Determine which cell (if any) a sprite is landing on
454         * 
455         * isLanding() is more efficient if the specific cell is not required
456         *
457         * @param Sprite sprite
458         *
459         * @return int  The cell the sprite is landing or -1 if none
460         */
461        public int landingCell(Sprite sprite) {
462            int spriceCentreX = sprite.getX() + (sprite.getWidth()/2) ;
463            int spriteCentreY = sprite.getY() + (sprite.getHeight()/2);
464    
465            for (int i=0 ; i<getColumns() ; i++) {
466                if (    spriteCentreY >= getY()
467                     && spriteCentreY < (getY() + getCellHeight())
468                     && spriceCentreX > (getX() + i * getCellWidth())
469                     && spriceCentreX <= (getX() + (i+1) * getCellWidth()) ) {
470                    
471                    return i;
472                }
473            }
474    
475            return -1;
476        }

484            // work out the index based on our current offset
485            int index = (x + currentOffset) % getColumns();

487            // get the cell contents
488            int content = cellContents[index];

490            return types[content];
