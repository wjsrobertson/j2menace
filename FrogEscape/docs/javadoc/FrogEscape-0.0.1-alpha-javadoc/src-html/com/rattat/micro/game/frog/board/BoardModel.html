605        /**
606         * Setup the board for beginning a new level
607         *
608         * @param int level
609         */
610        public void initLevel (int level) {
611            if (level > numLevels) {
612                notifyListeners (BoardEvent.GAME_WON);
613                return;
614            }
615            
616            currentLevel = level;
617            initFrogPosition ();
618            frogActive = true;
619            deadFrog.setVisible (false);
620            homeLayer.reset ();
621            statusLayer.setLives (numLives);
622            statusLayer.setTime (LEVEL_TIME);
623            createChannels ();
624            notifyListeners (BoardEvent.LEVEL_START);
625        }
626        
627        /**
628         * Update the board by one unit of time:
629         *
630         * Handle frog movement
631         *
632         * Update home layer, channels and status / score layers
633         *
634         * Check if the frog is dying
635         *
636         * Finish the game is the player is dead and the dead frog aninmation 
637         * is complete
638         */
639        public void tick () {
640            homeLayer.tick ();
641            
642            updateFrogPositionUser ();
643            updateDeadFrog ();
644            
645            for (int i=0 ; i<channels.length ; i++) {
646                channels[i].update ();
647                
648                if (channels[i].isLanding (frog)) {
649                    if ( channels[i].cellType ( channels[i].landingCell (frog) ) == Channel.TYPE_MOVING ) {
650                        frog.move ( channels[i].getVelocity (), 0 );
651                    }
652                }
653                
654                if (channels[i].isColliding (frog)) {
655                    if ( channels[i].isDeadlyCollision (frog) ) {
656                        System.out.println ( "Channel Frog Death" );
657                        frogDeath ();
658                    }
659                }
660            }
661            
662            if (frogActive && (frog.getX () < 0 || frog.getX () + frog.getWidth () > getWidth () ) ) {
663                frogDeath ();
664            }
665            
666            if ( statusLayer.getTime () == 0 ) {
667                frogDeath ();
668            }
669            
670            if (frogActive) {   // don't increment timer when frog inactive
671                statusLayer.tick ();
672            }
673            
674            // check if game is over
675            
676            if (numLives == 0 && deadFrogTimer > deadFrogLonjevity) {
677                finishGame ();
678            }
679        }

681        /**
682         * Handle loosing a life
683         *
684         * Hide the frog, repoisition him at the foot of the screen, deactivate him
685         * and mark the dead frog sprite as active
686         */
687        public void frogDeath () {
688            startDeadFrog();
689            initFrogPosition();
690            looseLife();
691            frog.setVisible (false);
692            deadFrogActive = true;
693            frogActive = false;
694        }
695        
696        /**
697         * 
698         */
699        private void updateDeadFrog () {
700            if (deadFrogActive) {
701                double frameRatio = deadFrogLonjevity / deadFrog.getFrameSequenceLength ();
702                
703                int frame = (int) Math.floor ( deadFrogTimer / frameRatio );
704                frame = Math.min ( deadFrog.getFrameSequenceLength ()-1, frame );
705                
706                if ( deadFrogTimer >= deadFrogLonjevity ) {
707                    deadFrogActive = false;
708                    deadFrog.setVisible (false);
709                    frog.setVisible (true);
710                    frogActive = true;
711                } else {
712                    deadFrog.setFrame (frame);
713                }
714                
715                deadFrogTimer++;
716            }
717        }
718        
719        private void startDeadFrog () {
720            int offsetX = (deadFrog.getWidth () - frog.getWidth ()) / 2; // offset from the frog position, not origin
721            int offsetY = (deadFrog.getHeight () - frog.getHeight ()) / 2; // offset from the frog position, not origin
722            
723            deadFrog.setPosition ( frog.getX () - offsetX, frog.getY () - offsetY );
724            deadFrog.setVisible (true);
725            deadFrogActive = true;
726            deadFrogTimer = 0;
727        }

852        public void homeEvent (int event, Home home) {
853            switch (event) {
854                case HomeEvent.FROG_DEATH:
855                    if (frogActive) {
856                        frogDeath ();
857                    }
858                    break;
859                    
860                case HomeEvent.FROG_HOME:
861                    initFrogPosition ();
862                    resetTime ();
863                    notifyListeners (BoardEvent.FROG_HOME);
864                    
865                    addScore (POINTS_FROG_HOME);
866                    
867                    if ( home.isComplete () ) {
868                        addScore (POINTS_LEVEL);
869                        notifyListeners (BoardEvent.LEVEL_COMPLETE);
870                        initLevel ( currentLevel+1 );
871                    }
872                    break;
873                    
874                case HomeEvent.INSECT_EATEN:
875                    addScore (POINTS_EAT_INSECT);
876                    break;
877            }
878        }

900         *
901         * @param int event
902         */
903        private void notifyListeners (int event) {
904            Enumeration enumeration = listeners.elements ();
905            while (enumeration.hasMoreElements ()) {

917            if (getFrog () != null && frogActive) {

924                        break;
925

928                            frog.setTransform (Sprite.TRANS_ROT90);
929                            setFrogMovement (FROG_RIGHT);
930                        }
931                        break;
932

935                            frog.setTransform (Sprite.TRANS_NONE);
936                            setFrogMovement (FROG_UP);
937                        }
938                        break;
939                        
940                    case USERINPUT_DOWN:

942                            if ( frog.getY () + 1 < 14*CELL_SIZE + (CELL_SIZE-frog.getWidth ())/2 ) {
943                                frog.setTransform (Sprite.TRANS_ROT180);
944                                setFrogMovement (FROG_DOWN);
945                            }
946                        }
947                        break;
948                }
949            }

953        /**
954         * update the position of the frog if the user is moving it
955         */
956        private void updateFrogPositionUser () {
957            if ( isFrogMoving () ) {
958                
959                int horizontalIncrement = 0;
960                int verticalIncrement   = 0;
961                int potentialIncrement  = 0;
962                int dx = 0;
963                int dy = 0;
964                
965                switch (getFrogMovement ()) {
966                    case BoardModel.FROG_UP:
967                        
968                        verticalIncrement = BoardModel.FROG_VERTICAL_INCREMENT;
969                        potentialIncrement = getFrogMoveCounter () + verticalIncrement;
970                        if ( potentialIncrement >= BoardModel.FROG_VERTICAL_TOTAL) {
971                            verticalIncrement = BoardModel.FROG_VERTICAL_TOTAL - getFrogMoveCounter ();
972                        }
973                        
974                        moveFrog (0, -verticalIncrement);
975                        addFrogMoveCounter (verticalIncrement);
976                        
977                        break;
978                        
979                    case BoardModel.FROG_DOWN:
980                        
981                        verticalIncrement = BoardModel.FROG_VERTICAL_INCREMENT;
982                        potentialIncrement = getFrogMoveCounter () + verticalIncrement;
983                        if ( potentialIncrement >= BoardModel.FROG_VERTICAL_TOTAL) {
984                            verticalIncrement = BoardModel.FROG_VERTICAL_TOTAL - getFrogMoveCounter ();
985                        }
986                        
987                        moveFrog (0, verticalIncrement);
988                        addFrogMoveCounter (verticalIncrement);
989                        
990                        break;
991                        
992                    case BoardModel.FROG_LEFT:
993                        
994                        horizontalIncrement = BoardModel.FROG_HORIZONTAL_INCREMENT;
995                        potentialIncrement = getFrogMoveCounter () + horizontalIncrement;
996                        if ( potentialIncrement >= BoardModel.FROG_HORIZONTAL_TOTAL) {
997                            horizontalIncrement = BoardModel.FROG_HORIZONTAL_TOTAL - getFrogMoveCounter ();
998                        }
999                        
1000                        moveFrog (-horizontalIncrement, 0);
1001                        addFrogMoveCounter (horizontalIncrement);
1002                        
1003                        break;
1004                        
1005                    case BoardModel.FROG_RIGHT:
1006                        
1007                        horizontalIncrement = BoardModel.FROG_HORIZONTAL_INCREMENT;
1008                        potentialIncrement = getFrogMoveCounter () + horizontalIncrement;
1009                        if ( potentialIncrement >= BoardModel.FROG_HORIZONTAL_TOTAL) {
1010                            horizontalIncrement = BoardModel.FROG_HORIZONTAL_TOTAL - getFrogMoveCounter ();
1011                        }
1012                        
1013                        moveFrog (horizontalIncrement, 0);
1014                        addFrogMoveCounter (horizontalIncrement);
1015                        
1016                        break;
1017                }
1018                
1019                if (verticalIncrement <= 0 && horizontalIncrement <= 0) {
1020                    if ( getFrogMovement () == BoardModel.FROG_UP && frog.getY () < uppermostYOrdinate ) {
1021                        uppermostYOrdinate = frog.getY ();
1022                        addScore (POINTS_JUMP);
1023                    }
1024                    setFrogMoveCounter (0);
1025                    setFrogMovement (BoardModel.FROG_NULL);
1026                }
1027            }
1028        }
