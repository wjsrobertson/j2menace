005    package com.rattat.micro.game.frog.board.elements;
006    
007    /**
008     * Very simple class - just holds public static final ints that 
009     * are used to notify BoardListeners of events. In most Observer pattern
010     * implementations an instance of the Event class would be passed around
011     * not notify of events, but creating all these objects is not a great thing in 
012     * MIDP. Passing around ints insatead is plenty effective. 
013     *
014     * @author william@rattat.com
015     */
016    public class HomeEvent {
017        
018        /**
019         * Frog dies from hitting the wrong bit of the home component
020         */
021        public static final int FROG_DEATH = 1;
022        
023        /**
024         * Frog has landed in an empty cell
025         */
026        public static final int FROG_HOME = 2;
027        
028        /**
029         * An insect has appeared in one of the cells
030         */
031        public static final int INSECT_ARRIVE = 3;
032        
033        /**
034         * An insect has left one of the cells
035         */
036        public static final int INSECT_LEAVE = 4;
037        
038        /**
039         * An aligator has appeared in one of the cells
040         */
041        public static final int ALIGATOR_ARRIVE = 5;
042        
043        /**
044         * An aligator has left one of the cells
045         */
046        public static final int ALIGATOR_LEAVE = 6;
047        
048        /**
049         * An insect has been eaten by the player's frog
050         */
051        public static final int INSECT_EATEN = 7;
052    
053        /** 
054         * Private constructor - should never be instanciated
055         */
056        private HomeEvent () {
057        }
058        
059    }
