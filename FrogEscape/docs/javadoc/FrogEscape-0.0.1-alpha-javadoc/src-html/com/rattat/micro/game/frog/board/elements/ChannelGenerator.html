465        /**
466         * Create a channel by channel identifier and level 
467         *
468         * @param int channelType
469         * @param int level
470         */
471        public Channel createChannel(int channelType, int level) throws Exception {
472            if (level < 1 || level > numLevels) {
473                throw new Exception("Invalid level");
474            }
475            
476            int velocity = 0;
477            int[] cellContents;
478            
479            level = level-1;    // 0-indexed internally
480            
481            switch (channelType) {
482                case CHANNEL_RIVER_0:
483                    velocity = velocities[CHANNEL_RIVER_0][level];
484                    cellContents = river0Cells[level];
485                    break;
486                    
487                case CHANNEL_RIVER_1:
488                    velocity = velocities[CHANNEL_RIVER_1][level];
489                    cellContents = river1Cells[level];
490                    break;
491                    
492                case CHANNEL_RIVER_2:
493                    velocity = velocities[CHANNEL_RIVER_2][level];
494                    cellContents = river2Cells[level];
495                    break;
496                    
497                case CHANNEL_RIVER_3:
498                    velocity = velocities[CHANNEL_RIVER_3][level];
499                    cellContents = river3Cells[level];
500                    break;
501                    
502                case CHANNEL_RIVER_4:
503                    velocity = velocities[CHANNEL_RIVER_4][level];
504                    cellContents = river4Cells[level];
505                    break;
506                    
507                case CHANNEL_ROAD_0:
508                    velocity = velocities[CHANNEL_ROAD_0][level];
509                    cellContents = road0Cells[level];
510                    break;
511                    
512                case CHANNEL_ROAD_1:
513                    velocity = velocities[CHANNEL_ROAD_1][level];
514                    cellContents = road1Cells[level];
515                    break;
516                    
517                case CHANNEL_ROAD_2:
518                    velocity = velocities[CHANNEL_ROAD_2][level];
519                    cellContents = road2Cells[level];
520                    break;
521                    
522                case CHANNEL_ROAD_3:
523                    velocity = velocities[CHANNEL_ROAD_3][level];
524                    cellContents = road3Cells[level];
525                    break;
526                    
527                case CHANNEL_ROAD_4:
528                    velocity = velocities[CHANNEL_ROAD_4][level];
529                    cellContents = road4Cells[level];
530                    break;
531                    
532                default:
533                    throw new Exception("Invalid channel type");
534            }
535            
536            return Channel.create(cellContents, velocity);
537            
538        }
