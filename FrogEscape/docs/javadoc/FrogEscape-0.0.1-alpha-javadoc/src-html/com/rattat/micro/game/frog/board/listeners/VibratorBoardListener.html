005    package com.rattat.micro.game.frog.board.listeners;
006    
007    import com.rattat.micro.game.frog.board.*;
008    import com.rattat.micro.ui.vibrator.Vibrator;
009    
010    /**
011     * vibrate the handset when the user looses a life
012     *
013     * implements BoardListener to know when to play the sound
014     *
015     * @author william@rattat.com
016     */
017    public class VibratorBoardListener implements BoardListener {
018        
019        /**
020         * Length of time to vibrate for when the player looses a life
021         */
022        private static final int DEATH_VIBRATION = 100;
023        
024        /**
025         * Creates a new instance of VibratorBoardListener
026         */
027        public VibratorBoardListener () {
028        }
029        
030        /**
031         * BoardListener interface method
032         *
033         * Listen for the BoardEvent.FROG_DEATH event and vibrate when it happens
034         *
035         * @param int event
036         * @param BoardModel model
037         */
038        public void boardEvent(int event, BoardModel model) {
039            switch (event) {
040                case BoardEvent.FROG_DEATH:
041                    Vibrator.vibrate(DEATH_VIBRATION);
042                    break;
043            }
044        }
045        
046    }
