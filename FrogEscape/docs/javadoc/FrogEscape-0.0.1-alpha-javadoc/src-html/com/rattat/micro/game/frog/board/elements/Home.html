025         * The number of free home cells

027        public final static int NUM_CELLS = 5;

030         * Represents an empty cell

037        public static final int INVALID_CELL = -2;

039        /**

041         */
042        public static final int FROG_HOME = 0;

045         * Represents a cell with a lurking aligator

050         * Represents a cell with a fully displayed aligator
051         */
052        public static final int ALIGATOR_2 = 2;

054        /**
055         * Represents a cell with an insect
056         */
057        public static final int INSECT = 3;

062        private String backgroundImageFile = "/__top.png";

108        /**
109         * Age of the currently displayed insect
110         */
111        private int insectTimer = 0;
112        
113        /**
114         * How active is the insect (how likely is he to appear) 0-100
115         */
116        private int insectActivity = 80;

118        /**

120         */
121        private int insectLongevity = 100;
122

124         * Which cell is the insect currently in

126        private int insectCell = NULL;
127            
128        /**
129         * Age of the currently displayed aligator

131        private int aligatorTimer = 0;
132        
133        /**
134         * How active is the aligator (how likely is he to appear) 0-100
135         */
136        private int aligatorActivity = 90;
137        
138        /**
139         * How many game ticks does the aligator live for

141        private int aligatorLongevity = 100;

144         * Which cell is the insect currently in

146        private int aligatorCell = NULL;

149         * How long the aligator will lurk for before he appears fully
150         */
151        private int aligatorLurkTime = 20;

154         * Used when determining whether or not to display an aligator or insect
155         */
156        private Random random = new Random();

158        /**
159         * Sprite to monitor - watch for collisions, landing on a home cell etc.

175        /**
176         * Initialise the home layer
177         */
178        private void init() {
179            try {
180                backgroundImage = Image.createImage(backgroundImageFile);
181                setWidth( backgroundImage.getWidth() );
182                setHeight( backgroundImage.getHeight() );
183                                       
184            } catch (Exception e) {
185                System.err.println("Error loading image" + e);
186            }
187            
188            initCellImages();
189            calculateOffsets();
190            reset();
191        }
192    
193        /**
194         * Initialise the images used when painting
195         */
196        private void initCellImages() {
197            for (int i=0 ; i<cellImages.length ; i++) {
198                try {
199                    cellImages[i] = Image.createImage(cellImageFiles[i]);
200                } catch (Exception e) {
201                    System.err.println("Error loading image (" + cellImages[i]  + "): " + e);
202                }
203            }
204        }

206        /**
207         * Reset the Home component to contain empty cells
208         */
209        public void reset() {
210            insectCell = NULL;
211            insectTimer = 0;
212            
213            aligatorCell = NULL;
214            aligatorTimer = 0;
215            
216            for (int i=0 ; i<cellContents.length ; i++) {
217                cellContents[i] = NULL;
218            }
219        }
220        
221        /**
222         * Initialise the offsets for the positions of the "home" cells
223         *
224         * These are the cells which the player is aiming to get the frog into
225         */
226        private void calculateOffsets() {        
227            cellWidth = cellImages[0].getWidth();
228    
229            cellYOffset = getY() + backgroundImage.getHeight() / 4;
230    
231            int minXOffset = getX() + backgroundImage.getHeight() / 4;
232            int seperatorDistance = (int) (minXOffset / 3) * 26;
233    
234            for (int i=0 ; i<cellXOffsets.length ; i++) {
235                cellXOffsets[i] = minXOffset + (i * seperatorDistance);
236            }
237        }
238        
239        /**
240         * Paint the Home component onto a graphics object
241         */
242        public void paint(Graphics g) {
243            calculateOffsets();
244            
245            g.drawImage(backgroundImage, getX(), getY(), Graphics.LEFT|Graphics.TOP);
246            
247            for (int i=0 ; i<cellContents.length ; i++) {
248                if (cellContents[i] != NULL) {
249                    g.drawImage(cellImages[ cellContents[i] ], cellXOffsets[i], cellYOffset, Graphics.TOP|Graphics.LEFT);
250                }
251            }
252        }

254        /**
255         * Check if all five slots have been filled with frog home's - i.e. the 
256         * player has got a from in each cell.
257         *
258         * @return boolean
259         */
260        public boolean isComplete() {
261            for (int i=0 ; i<cellContents.length ; i++) {
262                if (cellContents[i] != FROG_HOME) {
263                    return false;
264                }
265            }
266    
267            return true;
268        }

270        /**
271         * Randomly create aligators and insects as required
272         *
273         * Also, update aligators / insects
274         *
275         * Also, check for things happening to the player's frog
276         */
277        public void tick() {
278            checkSprite();
279            updateInsect();
280            updateAligator();
281        }
282        
283        /**
284         * Randomly create an insect if there isn't one
285         *
286         * Hide the insect if he has been around for long enough (insectLongevity)
287         */
288        private void updateInsect() {
289            if ( insectCell == NULL && 
290                 hasFreeCell() &&
291                 randomActionHappens(insectActivity) ) {
292                
293                insectCell = chooseRandomFreeCell();
294                
295                if (insectCell != NULL) {
296                    insectTimer = 0;
297                    cellContents[insectCell] = INSECT;
298                    notifyListeners(HomeEvent.INSECT_ARRIVE);
299                }
300            }
301            
302            if (insectTimer >= insectLongevity) {
303                if (insectCell != NULL && cellContents[insectCell] == INSECT) {
304                    cellContents[insectCell] = NULL;
305                }
306                insectCell = NULL;
307                insectTimer = 0;
308                notifyListeners(HomeEvent.INSECT_LEAVE);
309            }
310            
311            if (insectCell != NULL) {
312                insectTimer++;
313            }
314        }
315            
316        /**
317         * Update the aligator
318         *
319         * Randomly create an aligator if there isn't one
320         *
321         * Display lurking / fully displayed aligator according to 
322         * how long the aligator has been displayed for (aligatorLurkTime)
323         *
324         * Hide aligator if he has been around for long enough (aligatorLongevity)
325         */
326        private void updateAligator() {
327            
328            // randomly create an alligator if one doesn't exist
329            
330            if ( aligatorCell == NULL && 
331                 hasFreeCell() &&
332                 randomActionHappens(aligatorActivity) ) {
333                
334                aligatorCell = chooseRandomFreeCell();
335                
336                if (aligatorCell != NULL) {
337                    aligatorTimer = 0;
338                    cellContents[aligatorCell] = ALIGATOR_1;
339                    notifyListeners(HomeEvent.ALIGATOR_ARRIVE);
340                }
341            }
342            
343            if (aligatorTimer > aligatorLurkTime) {
344                cellContents[aligatorCell] = ALIGATOR_2;
345            }
346            
347            if (aligatorTimer >= aligatorLongevity) {
348                if (aligatorCell != NULL && cellContents[aligatorCell] == ALIGATOR_2) {
349                    cellContents[aligatorCell] = NULL;
350                }
351                aligatorCell = NULL;
352                aligatorTimer = 0;
353                notifyListeners(HomeEvent.ALIGATOR_LEAVE);
354            }
355            
356            if (aligatorCell != NULL) {
357                aligatorTimer++;
358            }
359        }

361        /**
362         * Randomly decide if an action is to happen or not
363         *
364         * @param int activity  Higher number means action is more likely to happen
365         */
366        private boolean randomActionHappens(int activity) {
367            if (activity <= 0 || numFreeCells() == 0) {
368                return false;
369            }
370            
371            /*
372             scale the activity down according to how many free cells there are
373             otherwise an alligator can appear over andover in one remaining free
374            */
375            int scaledActivity = (int) (activity * ((double)numFreeCells() / NUM_CELLS));
376            
377            int activityFactor = 1 + 10 * (100 - Math.min(scaledActivity, 100));
378            
379            return random.nextInt( activityFactor ) >= activityFactor-1;
380        }

403        /**
404         * Check if the Home instance has a free cell
405         */
406        private boolean hasFreeCell() {
407            for (int i=0 ; i<cellContents.length ; i++) {
408                if (cellContents[i] == NULL) {
409                    return true;
410                }
411            }
412            
413            return false;
414        }
415        
416        /**
417         * Get the number of free cells
418         */
419        private int numFreeCells() {
420            int num = 0;
421            
422            for (int i=0 ; i<cellContents.length ; i++) {
423                if (cellContents[i] == NULL) {
424                    num++;
425                }
426            }
427            
428            return num;
429        }

431        /**
432         * Randomly choose a free cell (a cell without anything in it)
433         *
434         * @return int
435         */
436        private int chooseRandomFreeCell() {   
437            int tmpCells[] = new int[cellContents.length];
438            int numFound = 0;
439            
440            for (int i=0 ; i<cellContents.length ; i++) {
441                if (cellContents[i] == NULL) {
442                    tmpCells[numFound] = i;
443                    numFound++;
444                }
445            }
446    
447            if ( numFound == 0 ) {
448                return -1;
449            } else if ( numFound == 1 ) {
450                return tmpCells[0];
451            }
452            
453            int randomIndex = random.nextInt(numFound);
454            
455            return tmpCells[randomIndex];
456        }

458        /**
459         * If there is a sprite to check, then check if it is landing on a cell.
460         *
461         * If it is landing on a cell, notify listeners of the relevant event
462         */
463        private void checkSprite() {
464            if (sprite != null) {
465                // if sprite isn't touching us, so we don't care about it
466                if (! isColliding(sprite)) {
467                    return;
468                }
469    
470                int landingCell = landingCell(sprite);
471    
472                if (landingCell == INVALID_CELL) { 
473                    notifyListeners(HomeEvent.FROG_DEATH);     
474                } else {
475                    switch ( getCellContents(landingCell) ) {
476                        case INSECT:   
477                            setCellContents(landingCell, FROG_HOME);
478                            notifyListeners(HomeEvent.INSECT_EATEN);
479                            notifyListeners(HomeEvent.FROG_HOME);
480                            break;
481    
482                        case NULL:
483                            setCellContents(landingCell, FROG_HOME);
484                            notifyListeners(HomeEvent.FROG_HOME);
485                            break;
486    
487                        case ALIGATOR_2:   
488                        case ALIGATOR_1: 
489                        case FROG_HOME: 
490                        default:
491                            notifyListeners(HomeEvent.FROG_DEATH);
492                            break;
493                    }
494                }
495            }
496        }

501         * @param Sprite sprite

503        public void setSprite(Sprite sprite) {
504            this.sprite = sprite;

507        /**
508         * Check if the sprite is colliding with this layer.
509         *
510         * Only checks y coordinates of the sprite since the home layer takes
511         * up the whole width of the screen
512         *
513         * @param Sprite sprite
514         *
515         * @return boolean
516         */
517        private boolean isColliding(Sprite sprite) {
518            return sprite.getY() <= (getY() + getHeight());
519        }
520        
521        /**
522         * Determine which cell a sprite is landing on (if any)
523         *
524         * returns INVALID_CELL if no cell is being landed on
525         *
526         * @param Sprite sprite
527         *
528         * @return int
529         */
530        private int landingCell(Sprite sprite) {       
531            for (int i=0 ; i<cellContents.length ; i++) {
532                try {
533                    if (isLanding(sprite, i)) {
534                        return i;
535                    }
536                } catch (Exception e) {
537                }
538            }
539            
540            return INVALID_CELL;
541        }
542        
543        /**
544         * Check if a sprite is landing on one of the cells
545         *
546         * A sprite is landing on one of the cells if it is fully within the 
547         * LHS and RHS of cell
548         *
549         * @param Sprite sprite     Sprite to check
550         * @param int cell          index of the cell to check
551         *
552         * @return boolean
553         */
554        private boolean isLanding(Sprite sprite, int cell) throws Exception {
555            if (cell <0 || cell >= cellXOffsets.length) {
556                throw new Exception("Invalid cell: " + cell);
557            }
558    
559            int cellMinX = cellXOffsets[cell];
560            int cellMaxX = cellMinX + cellWidth;
561    
562            if ( sprite.getY() <= (getY() + getHeight())
563                 && (sprite.getX() >= cellMinX)
564                 && ((sprite.getX() + sprite.getWidth()) <= cellMaxX )) {
565    
566                return true;
567            }
568    
569            return false;
570        }

572        /**
573         * Remove a listener. The listener will no longer be notified of Home events
574         *
575         * @param HomeListener listener
576         */
577        public void removeListener(HomeListener listener) {
578            listeners.removeElement(listener);
579        }
580        
581        /**
582         * Add a listener. The listener will be notified of Home events
583         *
584         * @param HomeListener listener
585         */
586        public void addListener(HomeListener listener) {
587            listeners.addElement(listener);
588        }
589        
590        /**
591         * Notify listeners of an event
592         *
593         * @param int event
594         */
595        private void notifyListeners(int event) {
596            Enumeration enumeration = listeners.elements();
597            while (enumeration.hasMoreElements()) {
598                HomeListener listener = (HomeListener) enumeration.nextElement();
599                listener.homeEvent(event, this);
600            }
601        }
