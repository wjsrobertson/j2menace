005    package com.rattat.micro.game.frog.board;
006    
007    import com.rattat.micro.game.frog.board.BoardModel;
008    
009    /**
010     * BoardListener interface should be implemented by objects wishing to be 
011     * notified of board events. e.g. new game, lost life etc.
012     *
013     * @author william@rattat.com
014     */
015    public interface BoardListener {
016        
017        /**
018         * Called when a BoardEvent happens
019         * 
020         * @param int event The releveant BoardEvent static int representing
021         *                  the event
022         * @param BoardModel model  The model the event happened on
023         */
024        public void boardEvent(int event, BoardModel model);
025    }
