005    package com.rattat.micro.game.frog;
006    
007    import javax.microedition.lcdui.Canvas;
008    import javax.microedition.lcdui.Graphics;
009    import javax.microedition.lcdui.Font;
010    
011    /**
012     * A simple class to display a page saying congrats for winning the game
013     *
014     * @author william@rattat.com
015     */
016    public class WonCanvas extends Canvas {
017    
018        /**
019         * title text
020         */
021        String title = "Well Done - you win!";
022        
023        /**
024         * little bit of congratulatory text so the user doesn't feel like 
025         * playing the game for so long was a complete waste of time.
026         */
027        String text = "The frogs love you :-)";
028    
029        /**
030         * Left hand side of a bit of text telling the user to press a key to continue
031         * in the middle of this and continueTextRight will be the name of the button 
032         * they have to press to continue
033         */
034        String continueTextLeft = "press ";
035        
036        /**
037         * Right hand side of a bit of text telling the user to press a key to continue
038         * in the middle of this and continueTextLeft will be the name of the button 
039         * they have to press to continue
040         */
041        String continueTextRight = " to continue";
042    
043        /**
044         * Colour of the text
045         */
046        int textColor = 0x00AA55;
047    
048        /**
049         * Background colour of the screen
050         */
051        int backgroundColor = 0x000000;
052        
053        /**
054         * Holds the Font object returned from the Graphics object - needed
055         * so that we can determine its height for spacing when painting on the 
056         * screen
057         */
058        Font font;
059        
060        /**
061         * Used to switch between different screens - needed to display the 
062         * main menu when the user continues from this screen
063         */
064        private ScreenChooser screenChooser;
065    
066        /**
067         * Creates a new instance of WonCanvas
068         */
069        public WonCanvas (ScreenChooser screenChooser) {
070            setFullScreenMode(true);
071            this.screenChooser = screenChooser;
072        }
073    
074        /**
075         * Canvas concrete method
076         *
077         * paint the congrats to the screen. 
078         *
079         * @param Graphics g
080         */
081        public void paint(Graphics g) {
082            // background
083            
084            g.setColor(backgroundColor);
085            g.fillRect(0, 0, getWidth(), getHeight());
086            
087            g.setColor(textColor);
088    
089            font = g.getFont();
090            
091            int yPos = getHeight() / 2 - font.getHeight()*2;
092            int xPos = getWidth() / 2;
093            g.drawString(title, xPos, yPos, Graphics.TOP|Graphics.HCENTER);
094            
095            yPos = getHeight() / 2;
096            xPos = getWidth() / 2;
097            g.drawString(text, xPos, yPos, Graphics.TOP|Graphics.HCENTER);
098            
099            String continueText = continueTextLeft
100                                  + getKeyName( getKeyCode(Canvas.FIRE) ) 
101                                  + continueTextRight;
102            
103            yPos = getHeight() / 2 + font.getHeight()*2;
104            xPos = getWidth() / 2;
105            g.drawString(continueText, xPos, yPos, Graphics.TOP|Graphics.HCENTER);
106        }
107        
108        /**
109         * Canvas method
110         * 
111         * Listen for the continue keypress - the Canvas.FIRE keypress
112         *
113         * @param int keyCode
114         */
115        public void keyPressed(int keyCode) {
116            if (getGameAction(keyCode) == Canvas.FIRE) {
117                try {
118                    screenChooser.displayScreen(ScreenChooser.SCREEN_MAIN_MENU);
119                } catch (Exception e) {
120                }
121            }
122            
123        }
124    }
