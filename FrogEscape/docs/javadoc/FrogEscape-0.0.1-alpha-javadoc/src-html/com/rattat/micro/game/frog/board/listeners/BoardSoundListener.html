005    package com.rattat.micro.game.frog.board.listeners;
006    
007    import com.rattat.micro.game.frog.board.*;
008    import com.rattat.micro.game.frog.sound.SoundManager;
009    
010    /**
011     * Listens for BoardEvents and plays sounds at the appropriate times
012     *
013     * @author william@rattat.com
014     */
015    public class BoardSoundListener implements BoardListener {
016    
017        /**
018         * The SoundManager object used to play frogger sounds
019         */
020        private SoundManager soundPlayer;
021    
022        /**
023         * Creates a new instance of BoardSoundListener
024         */
025        public BoardSoundListener () {
026            soundPlayer = SoundManager.getInstance();
027        }
028    
029        /**
030         * BoardListener interface method
031         *
032         * Listen for BoardEvent.LEVEL_START, BoardEvent.FROG_HOME, BoardEvent.FROG_DEATH
033         * and pay the appropriate dounds when these event happen
034         */
035        public void boardEvent(int event, BoardModel board) {
036            switch (event) {
037                case BoardEvent.GAME_START:
038                    //soundPlayer.play(SoundManager.SOUND_NEW_GAME);
039                    break;
040                    
041                case BoardEvent.LEVEL_START:
042                    soundPlayer.play(SoundManager.SOUND_NEW_LEVEL);
043                    break;
044    
045                case BoardEvent.FROG_HOME:
046                    soundPlayer.play(SoundManager.SOUND_FROG_HOME);
047                    break;
048                    
049                case BoardEvent.FROG_DEATH:
050                    soundPlayer.play(SoundManager.SOUND_FROG_DEATH);
051                    break;
052                                    
053            }
054        }
055    }
