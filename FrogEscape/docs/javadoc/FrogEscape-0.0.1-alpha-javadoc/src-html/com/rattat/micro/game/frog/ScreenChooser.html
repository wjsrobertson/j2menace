005    package com.rattat.micro.game.frog;
006    
007    import com.rattat.micro.game.frog.*;
008    
009    /**
010     * Simple class for choosing which screen to display. Rather than passing
011     * around a midlet instance, pass this around instead. Safer.
012     *
013     * @author william@rattat.com
014     */
015    public class ScreenChooser {
016        
017        /**
018         * The FroggerMIDLet instance
019         */
020        Midlet midlet;
021        
022        /**
023         * The main menu
024         */
025        public static final int SCREEN_MAIN_MENU = 1;
026        
027        /**
028         * The game board
029         */
030        public static final int SCREEN_BOARD = 2;
031        
032        /**
033         * The pause screen (main menu with extra option)
034         */
035        public static final int SCREEN_PAUSE = 3;
036        
037        /**
038         * The game won screen
039         */
040        public static final int SCREEN_WON = 4;   
041    
042        /** 
043         * Creates a new instance of ScreenChooser 
044         *
045         * Takes FroggerMIDLet as a parameter - used for actually
046         * displaying the appropriate Displayables
047         *
048         * @param Midlet midlet
049         */
050        public ScreenChooser (Midlet midlet) {
051            this.midlet = midlet;
052        }
053    
054        /**
055         * Display a particular screen must be one of the SCREEN_XXX variables
056         *
057         * @param int name  Name of screen to display
058         *
059         * @throws Exception if invalid parameter is passed in (i.e. undefined 
060         * screen - not SCREEN_XXX) 
061         */
062        public void displayScreen(int name) throws Exception {
063            switch (name) {
064                case SCREEN_MAIN_MENU:
065                    midlet.displayMainMenu();
066                    break;
067                    
068                case SCREEN_BOARD:
069                    midlet.displayBoard();
070                    break;
071                    
072                case SCREEN_PAUSE:
073                    midlet.displayPause();
074                    break;
075                    
076                case SCREEN_WON:
077                    midlet.displayWon();
078                    break;
079            
080                default:
081                    throw new Exception("Invalid screen");
082            }
083        }    
084    }
