014    /**
015     * Used to paint a model to an image
016     *
017     * Actually, this is pointles & a waste of resources. Need to do away
018     * with this altogether.
019     *
020     * @todo refactor to do painting in model
021     *
022     * @author william@rattat.com
023     */
024    public class BoardView {
025        
026        /**
027         * The model - the data representation of the game state
028         */
029        BoardModel model;
030        
031        /**
032         * Rendered Image representation of the model returned to the GameCanvas 
033         */
034        Image image = null;
035        
036        /**
037         * Graphics component of the image - used for painting
038         */
039        Graphics graphics = null; 
040    
041        /**
042         * Creates a new instance of BoardView
043         *
044         * @param BoardModel model
045         */
046        public BoardView (BoardModel model) {
047            this.model = model;
048            init();
049        }
050            
051        /**
052         * Initialise the model, ready for use
053         */
054        private void init() {
055            image    = Image.createImage( model.getWidth(), model.getHeight() );
056            graphics = image.getGraphics();
057        }
058        
059        /**
060         * Paint the game to the screen based on model data
061         *
062         * @return Image
063         */
064        public Image render() {
065            graphics.setColor(255, 255, 255); // white
066            graphics.fillRect(0, 0, model.getWidth(), model.getHeight());
067    
068            model.getScoreLayer().paint(graphics);
069            model.getStatusLayer().paint(graphics);
070            model.getHomeLayer().paint(graphics);
071            
072            for (int i=0 ; i<model.getChannels().length ; i++) {
073                try {
074                    model.getChannel(i).paint(graphics);
075                } catch (Exception e) {
076                    System.err.println("Error painting channel: " + e);
077                }
078            }
079            
080            for (int i=0 ; i<model.getBorderLayers().length ; i++) {
081                model.getBorderLayers()[0].paint(graphics);
082                model.getBorderLayers()[1].paint(graphics);
083            }
084    
085            model.getFrog().paint(graphics);
086            model.getDeadFrog().paint(graphics);
087            
088            return image;
089        }
090    }
