015    /**
016     * <p>
017     * A collection of static methods providing basic functionality for dealing with 
018     * arrays.
019     * </p>
020     *
021     * <p>
022     * This class is something like the great 
023     * <a href="http://jakarta.apache.org/commons/lang/api/org/apache/commons/lang/StringUtils.html">Apache Commons StringUtils library</a>
024     * for j2se. and aims to be the stripped-down equivalent for j2me.
025     * </p>
026     *
027     * <p>Due to size limitations, only methods for dealing with int, 
028     * double and Object arrays are included since these are the most commonly used. 
029     * String arrays can be dealt with by using the Object methods and casting 
030     * the return type where required.
031     * </p>
032     *
033     * <p>
034     * This class can't be instanciated.
035     * </p>
036     *
037     * @author william@rattat.com
038     */
039    public class ArrayUtils {

041        /**
042         * Random instance used to shuffle arrays

044        private static Random random = null;

050        /**
051         * Check if an integer value is in an integer array
052         *
053         * @param array The array to check
054         * @param value The value to search for
055         *
056         * @return true if value is in array, false otherwise
057         */
058        public static boolean contains( int[] array, int value) {
059            return indexOf(array, value) != -1;
060        }

086        /**
087         * <p>Get the index of a value in an array</p>
088         *
089         * <p>Returns -1 if the integer is not in the array</p>
090         *
091         * @param array The array to check
092         * @param value The value to search for
093         *
094         * @return  index or -1 if not found
095         *
096         * @throw   NullPointerException if array is null
097         */
098        public static int indexOf( int[] array, int value) {
099            if (array == null) {
100                throw new NullPointerException("array can't be null");
101            }
102            
103            for ( int i=0 ; i<array.length ; i++ ) {
104                if ( array[i] == value ) {
105                    return i;
106                }
107            }
108    
109            return -1;
110        }

164        /**
165         * Shuffle the values of an int array into random order
166         *
167         * @param array The array to shuffle
168         */
169        public static void shuffle(int[] array) {
170            if (array == null) {
171                return;
172            }
173            
174            for (int i=0 ; i<array.length ; i++) {
175                int r = i + random.nextInt( array.length-i );
176                
177                int tmp = array[i];
178                array[i] = array[r];
179                array[r] = tmp;
180            }
181        }

221        /**
222         * Append one integer array to another
223         *
224         * Returns a new array. Either array may be null. If both are null
225         * then null is returned.
226         *
227         * @param array1    First array
228         * @param array2    Array to append to first
229         *
230         * @return  The new array
231         */
232        public static int[] append(int[] array1, int[] array2) {
233            
234            if (array1 == null && array2 == null) {
235                return null;
236            }
237            if (array1 == null) {
238                return clone( array2 );
239            }
240            if (array2 == null) {
241                return clone( array1 );
242            }
243            
244            int length1 = array1.length;
245            int length2 = array2.length;
246            int[] newArray = new int[length1+length2];
247            
248            for ( int i=0 ; i<length1 ; i++ ) {
249                newArray[i] = array1[i];
250            }
251            for ( int i=0 ; i<length1 ; i++ ) {
252                int index = length1 + i;
253                newArray[index] = array1[i];
254            }
255            
256            return newArray;
257        }

335        /**
336         * Return a new copy of a double aray holding the same elements
337         *
338         * @param array    Array to append to clone
339         *
340         * @return  The cloned array
341         */
342        public static double[] clone(double[] array) {
343            if (array == null) {
344                return null;
345            }
346            
347            double[] clone = new double[array.length];
348            
349            for (int i=0 ; i<array.length ; i++) {
350                clone[0] = array[i];
351            }
352            
353            return clone;
354        }
