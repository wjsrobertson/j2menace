013    package com.rattat.micro.game.mb.mvc.board;
014    
015    import javax.microedition.lcdui.Graphics;
016    import javax.microedition.lcdui.Image;
017    import javax.microedition.lcdui.game.TiledLayer;
018    
019    import com.rattat.micro.game.mb.Block;

021    /**
022     * A class for rendering a {@code BoardModel} onto a 
023     * {@code Graphics} instnace
024     *
025     * @author william@rattat.com
026     */
027    public class BoardView
028     {
029    
030        /**
031         * The layer representing the board
032         */
033        private TiledLayer background;
034    
035        /**
036         * The layer representing the block that the user is 
037         * currently controlling
038         */
039        private TiledLayer userBlock;
040    
041        /**
042         * The model of the board
043         */
044        private BoardModel model;
045    
046        /**
047         * The width in pixels of one cell
048         */
049        private int cellSize;
050        
051        /**
052         * The x ordinate of the top-left corner where this
053         * view will be rendered when painted
054         */
055        private int x;
056        
057        /**
058         * The y ordinate of the top-left corner where this
059         * view will be rendered when painted
060         */
061        private int y;
062        
063        /**
064         * The width of the rendered board in pixels
065         */
066        private int width = 0;
067        
068        /**
069         * The height of the rendered board in pixels
070         */
071        private int height = 0;
072        
073        /**
074         * The background colour of the board
075         */
076        private int backgroundColor = 0x0000AA;
077        
078        /*
079         * The indexes of the TiledLayer image cells for each tetrad
080         */
081        /*

089            // TETRAD_O
090            4,
091            // TETRAD_S
092            5,

094            6,

098    
099        /**
100         * <p>
101         * Paint the board
102         * </p>
103         *
104         * <p>
105         * This method just delegates to the {@code TiledLayer} instances
106         * </p>
107         * 
108         * @param g
109         */
110        public void paint(Graphics g) {
111            // paint the background of the board in the appropriate position
112            
113            updateBoardLayer();
114            updateBlockLayer();
115            
116            g.setColor(backgroundColor);
117            g.fillRect(x, y, width, height);
118            
119            background.paint(g);
120            userBlock.paint(g);
121        }
122    
123    
124        /**
125         * <p>
126         * Creates a new instance of BoardView   
127         * </p>
128         * 
129         * <p>
130         * The {@code tiles} parameter is an image which is used to
131         * build {@code TiledLayer} instances which are used for 
132         * rendering the {@code BoardModel}. The image must contain
133         * eight differnet square cells, each of which is used to colour
134         * one tetrad type. 
135         * </p>
136         * 
137         * <p>
138         * The width & height of a cell is defined by the height of the {@code tiles}
139         * Image. Therefore, the width and height (in pixels) of the full board is 
140         * defined to be the height of {@code tiles} x num cells high and the width 
141         * is defined to be {@code tiles} x num cells wide.
142         * </p>
143         * 
144         * @param model     The model that will be rendered
145         * @param tiles     An image containing the tiles
146         */
147        public BoardView (BoardModel model, Image tiles) {
148            this.model = model;
149    
150            cellSize = tiles.getHeight();
151    
152            background = new TiledLayer(model.getNumCellsWide(), 
153                                        model.getNumCellsHigh(), 
154                                        tiles, cellSize, cellSize);
155    
156            userBlock = new TiledLayer(4, 4, tiles, cellSize, cellSize);
157            
158            width  = (model.getNumCellsWide()) * cellSize;
159            height = (model.getNumCellsHigh()) * cellSize;
160        }
161    
162        /**
163         * Update the block shape to represent the current tetrad and rotation
164         */
165        private void updateBlockLayer() {
166            boolean[][] shape = Block.getShape( model.getTetrad(), model.getRotation() );
167            
168            // get the index of the tile in the TiledLayer
169            // which is correct for the tetrad that is in play.
170            int tileIndex = model.getTetrad() + 1; //tetradImageIndexes[model.getTetrad()];
171            
172            for (int i=0 ; i<4 ; i++) {
173                for (int j=0 ; j<4 ; j++) {
174                    if (shape[i][j]) {
175                        userBlock.setCell(i, j, tileIndex);
176                    } else {
177                        userBlock.setCell(i, j, 0);
178                    }
179                }
180            }
181            
182            userBlock.setPosition( cellSize*model.getBlockCellX(), 
183                                   cellSize*model.getBlockCellY() );
184        }
185        
186        /**
187         * 
188         */
189        private void updateBoardLayer() {
190            for (int y=0 ; y<model.getNumCellsHigh() ; y++) {
191               for (int x=0 ; x<model.getNumCellsWide() ; x++) {
192                   if (model.getCellContents(x, y) == BoardModel.BLOCK_NULL) {
193                       background.setCell(x, y, 0);
194                   } else if (model.getCellContents(x, y) != BoardModel.BLOCK_OFFSCREEN ) {
195                        try {
196                            background.setCell(x, y, model.getCellContents(x, y) + 1);
197                        } catch (Exception e) {
198                        }
199                    }
200                }
201            }
202        }
203        
204        /**
205         * Set the x,y pixel position of the view
206         *
207         * @param x
208         * @param y
209         */
210        public void setPosition(int x, int y) {
211            this.x = x;
212            
213            this.y = y;
214        }
215        
216        /**
217         * Get the width of the full board area in pixels
218         * 
219         * @return
220         */
221        public int getWidth() {
222            return width;
223        }
224        
225        /**
226         * Get the height of the full board area in pixels
227         * 
228         * @return
229         */
230        public int getHeight() {
231            return height;
232        }
233    }
