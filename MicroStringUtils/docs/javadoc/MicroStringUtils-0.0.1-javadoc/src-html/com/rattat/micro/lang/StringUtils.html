015    /**
016     * <p>
017     * A collection of static methods providing basic functionality for dealing with 
018     * strings.
019     * </p>
020     *
021     * <p>
022     * This class is modeled after the fantastic 
023     * <a href="http://jakarta.apache.org/commons/lang/api/org/apache/commons/lang/StringUtils.html">Apache Commons StringUtils library</a>
024     * and aims to be the stripped-down equivalent for j2me. Although the implementation 
025     * is different, the methods do try to here adhere to roughly the same contracts 
026     * as the AC versions. This includes the desirable behaviour of dealing with nulls 
027     * in a friendly and intuitive fashion.
028     * </p>
029     *
030     * <p>
031     * To keep the compiled class as small as possible, the StringUtils 
032     * does not contain any means of dealing with characters directly - only strings.
033     * To use this API with characters, you should convert the character to a string first
034     * using <code>String.valueOf(char c)</code>. 
035     * </p>
036     *
037     * <p>
038     * e.g. to do a search and replace with characters you should do: 
039     * </p>
040     *
041     * <p>
042     * <code>StringUtils.replace("test", String.valueOf('s'), String.valueOf('n'))</code>
043     * </p>
044     *
045     * <p>
046     * This class can't be instanciated.
047     * </p>
048     *
049     * @author william@rattat.com
050     */
051    public class StringUtils {
052        
053        /**
054         * Left hand side of a string
055         */
056        public static final int LEFT = 1;
057        
058        /**
059         * Right hand side of a string
060         */
061        public static final int RIGHT = 2;
062        
063        /**
064         * Centre of a string
065         */
066        public static final int CENTER = LEFT | RIGHT;
067        
068        /**
069         * Left and right hand side of a string
070         */
071        public static final int BOTH = LEFT | RIGHT;

080        /**
081         * <p>Split a string into a String[] array based on another string to
082         * split by.</p>
083         * 
084         * <p>
085         * e.g. The following will produce a 3 element array with the elements "1", 
086         * "2" and "3"
087         * </p>
088         * 
089         * <p>
090         * <code>
091         * String[] pieces = StringUtils.split("1:2:3", ":");<br />
092         * </code>
093         * </p>
094         * 
095         * <p>
096         * If <code>splitBy</code> is longer than the string to split by then
097         * a single element array will be returned containing the original string.
098         * </p>
099         *
100         * <p>
101         * <code>splitBy</code> may be null or an empty 
102         * string - in which case the result is identical to that returned by 
103         * {@link StringUtils#split(String toSplit)}. i.e. The string will be split
104         * into individual strings, one for each character.
105         * </p>
106         *
107         * <p>
108         * <code>toSplit</code> parameter may be null - in which case the response 
109         * is null.
110         * </p>
111         * 
112         * <p>
113         * <code>toSplit</code> may be an empty 
114         * string - in which case the result is an empty array.
115         * </p>
116         * 
117         * @param   toSplit    The String to split up into pieces
118         * @param   splitBy     The String to split by
119         *
120         * @return  A String[] array composed of parts of the tosplit parameter
121         *
122         * @see StringUtils#split(String toSplit)
123         * @see StringUtils#join(String[] toJoin, String glue)
124         *
125         * @since 0.1
126         *
127         * @todo - optimise - do away with Vector
128         */
129        public static String[] split(String toSplit, String splitBy) {
130            
131            if (toSplit == null) {
132                return null;
133            }
134            if (splitBy == null) {
135                return split(toSplit);
136            }  
137            if (toSplit.equals("")) {
138                return new String[]{};
139            }
140            if (splitBy.equals("")) {
141                return split(toSplit);
142            }
143            if (toSplit.length() <= splitBy.length()) {
144                return new String[]{toSplit};
145            }
146    
147            // perform calculation
148            
149            Vector parts = new Vector();
150            int start = 0;
151            int end   = 0;
152            String remainder = toSplit;
153    
154            do {
155                end = remainder.indexOf(splitBy);
156                if ( end < 0 ) {
157                    break;
158                } else {
159                    parts.addElement( remainder.substring( 0, end ) );
160                    remainder = remainder.substring( end + splitBy.length() );
161                }
162            } while ( true );
163            
164            if ( remainder.length() >= 0 ) {
165                parts.addElement( remainder );
166            }
167    
168            String[] splitVals = new String[ parts.size() ];
169            parts.copyInto(splitVals);
170            
171            return splitVals;
172        }

174        /**
175         * <p>Split a string into an array composed of the string of each character.</p>
176         *
177         * <p>
178         * <code>toSplit</code> can be null in which case the response is null
179         * </p>
180         *
181         * <p>
182         * <code>toSplit</code> may be an empty string in which case 
183         * the result is an empty array.
184         * </p>
185         *
186         * @param toSplit    The String to split into individual character strings
187         *
188         * @return  A String[] array containing each individual character of toSplit
189         *
190         * @see StringUtils#split(String toSplit, String splitBy)
191         * @see StringUtils#join(String[] toJoin, String glue)
192         *
193         * @since 0.1
194         */
195        public static String[] split(String toSplit) {
196            
197            // fail-fast with null pointer exception
198            
199            if (toSplit == null) {
200                return null;
201            }
202            
203            // return empty array if string is empty
204            
205            if (toSplit.equals("")) {
206                return new String[]{};
207            }
208            
209            char[] chars = toSplit.toCharArray();
210            String[] strings = new String[chars.length];
211            
212            for (int i=0 ; i<chars.length ; i++) {
213                strings[i] = toSplit.valueOf( chars[i] );
214            }
215            
216            return strings;
217        }

219        /**
220         * <p>Join an  array of strings using glue.</p> 
221         *
222         * <p>
223         * i.e. concatenate an array of strings into one string consisting of the indiual
224         * strings in sequence, seperated by the glue string.
225         * </p>
226         *
227         * <p>
228         * e.g. The following will produce the string <code>"a@b@c@d"</code>
229         * </p>
230         * 
231         * <p>
232         * <code>
233         * String[] pieces = {"a","b","c","d"};<br />
234         * String joined = StringUtils.join(pieces, "@");
235         * </code>
236         * </p>
237         *
238         * <p>
239         * <code>toJoin</code> may be an empty array - in which case
240         * the response will be an empty string. 
241         * </p>
242         * 
243         * <p>
244         * <code>toJoin</code> may
245         * be null in ehich case the response is null. 
246         * </p>
247         * 
248         * <p>
249         * <code>glue</code> may be null or an empty 
250         * string - in which case the result is a single string containing
251         * the sequence of strings in the toJoin array.
252         * </p>
253         *
254         * 
255         * @param toJoin    A String[] array containing strings to join
256         * @param glue      The string to join with
257         *
258         * @return  The joined String
259         *
260         * @see StringUtils#split(String toSplit)
261         * @see StringUtils#split(String toSplit, String splitBy)
262         *
263         * @since 0.1
264         */
265        public static String join(String[] toJoin, String glue) {
266    
267            // fail-fast with null pointer exceptions
268            
269            if (toJoin == null) {
270                return null;
271            }
272            if (glue == null) {
273                glue = "";
274            }
275            
276            // special case for speed
277            
278            if ( toJoin.length == 0 ) {
279                return "";
280            }
281            
282            StringBuffer buffer = new StringBuffer();
283            
284            for (int i=0 ; i<toJoin.length ; i++) {
285                buffer.append( toJoin[i] );
286                
287                if ( i != toJoin.length-1 ) {
288                    buffer.append(glue);
289                }
290            }
291            
292            return buffer.toString();
293        }

295        /**
296         * <p>Replace substrings of a string with another string.</p>
297         *
298         * <p>
299         * e.g. The following will produce the string <code>"there 
300         * is nothing there"</code>
301         * </p>
302         * 
303         * <p>
304         * <code>
305         * String replaced = StringUtils.replace("it is nothing it", "it", "there");
306         * </code>
307         * </p>
308         *
309         * <p>
310         * If <code>subject</code> is null then null will be returned.
311         * </p>
312         *
313         * <p>
314         * If <code>search</code> is null or empty then the value of 
315         * <code>toReplace</code> will be returned.
316         * </p>
317         *
318         * <p>
319         * If the <code>replace</code> parameter is null it will be treated as an 
320         * empty string.
321         * </p>
322         *
323         * @param subject   The string in which to perform the search/replace
324         * @param search    The string to search for
325         * @param replace   The string to replace by
326         *
327         * @return
328         *
329         * @since 0.1
330         */
331        public static String replace(String subject, String search, String replace) {
332            
333            // fail-fast with null pointer exceptions
334            
335            if (subject == null) {
336                return null;
337            }
338            if (search == null || search.equals("")) {
339                return subject;
340            }
341            if (replace == null) {
342                replace = "";
343            }
344            
345            String[] split = split(subject, search);
346            
347            return join(split, replace);
348        }

351         * <p>Reverse a string.</p>
352         * 
353         * <p>Reverse the sequence of characters in the string.</p>
354         *
355         * <p>
356         * e.g. The following will produce the string <code>"edcba"</code>
357         * </p>
358         * 
359         * <p>
360         * <code>
361         * String reversed = StringUtils.reverse("abcde");
362         * </code>
363         * </p>
364         *
365         * <p>
366         * <code>toReverse</code> may not be null, but can be an empty 
367         * string
368         * </p>
369         *
370         * @param toReverse The string to reverse
371         *
372         * @return String   The reversed string 
373         *
374         * @since 0.1
375         */
376        public static String reverse( String toReverse ) {
377    
378            if (toReverse == null) {
379                return null;
380            }
381            if ( toReverse.equals("") ) {
382                return "";
383            }
384            
385            StringBuffer buffer = new StringBuffer();
386            
387            for ( int i=toReverse.length()-1 ; i >= 0 ; i-- ) {
388                buffer.append( toReverse.charAt(i) );
389            }
390            
391            return buffer.toString();
392        }

394        /**
395         * <p>
396         * Strip characters from the left hand side, right hand side or both 
397         * sides of a string.
398         * </p>
399         *
400         * <p>
401         * Which sides will be trimmed is decided by the <code>side</code> parameter.
402         * Legal values are <code>LEFT, RIGHT or BOTH</code>
403         * </p>
404         *
405         * <p>
406         * e.g. 1) The following will produce the string <code>"las"</code>
407         * </p>
408         * 
409         * <p>
410         * <code>
411         * String trimmed = StringUtils.trim("glasgow", "gow", StringUtils.BOTH);
412         * </code>
413         * </p>
414         *
415         * <p>
416         * e.g. 2) The following will produce the string <code>"dund"</code>
417         * </p>
418         * 
419         * <p>
420         * <code>
421         * String trimmed = StringUtils.trim("dundee", "e", StringUtils.RIGHT);
422         * </code>
423         * </p>
424         *
425         * @param toTrim    The string to trim
426         * @param chars     A sequence of characters to trim 
427         * @param side      Side to trim - LEFT, RIGHT or BOTH
428         *
429         * @return String   The trimmed string
430         *
431         * @since 0.1
432         */
433        public static String trim(String toTrim, String chars, int side) {
434    
435            if ( ((RIGHT & side) | (LEFT & side)) == 0 ) {
436                throw new IllegalArgumentException("side must be LEFT, RIGHT or BOTH");
437            }
438            if (toTrim == null) {
439                return null;
440            }
441            if (chars == null) {
442                return toTrim;
443            }
444            
445            // handle special case of empty string
446            
447            if (toTrim.length() == 0) {
448                return "";
449            }
450            
451            // handle special case of empty chars
452            
453            if (chars.length() == 0) {
454                return toTrim;
455            }
456            
457            // do the trim
458            
459            int first = 0;
460            int last = toTrim.length() - 1;
461            
462            // left hand side
463            
464            if ( (side | LEFT) > 0 ) {
465                for (int i=0 ; i<toTrim.length() ; i++ ) {
466    
467                    first = i;
468                    String testChar = toTrim.substring(i,i+1);
469    
470                    if ( chars.indexOf(testChar) == -1 ) {
471                        break;
472                    }
473                }
474            }
475            
476            // right hand side
477            
478            if ( (side | RIGHT) > 0 ) {
479                for (int i=toTrim.length()-1 ; i>=0 ; i-- ) {
480    
481                    last = i;
482                    String testChar = toTrim.substring(i,i+1);
483    
484                    if ( chars.indexOf(testChar) == -1 ) {
485                        break;
486                    }           
487                }
488            }
489            
490            return toTrim.substring(first, last+1);
491        }

524        /**
525         * Add padding to either the left hand side, right hand side or both 
526         * sides of a string.
527         *
528         * @param toPad     The string to add padding to 
529         * @param padding   The string to pad by
530         * @param size      The final length of the padded string
531         * @param side      The side to pad - LEFT, RIGHT or CENTER
532         *
533         * @return  The padded string
534         *
535         * @todo optimise and tidy
536         */
537        public static String pad(String toPad, String padding, int size, int side) {
538    
539            if ( ((RIGHT & side) | (LEFT & side)) == 0 ) {
540                throw new IllegalArgumentException("side must be LEFT, RIGHT or CENTER");
541            }
542            if (padding == null || padding.length() == 0) {
543                return toPad;
544            }
545            
546            size = Math.max(0, size);   // negative numbers are zero
547            
548            int totalPaddingSize = 0;
549            
550            if (toPad == null) {
551                return repeat( padding, size );
552            } else {
553                totalPaddingSize = size - toPad.length();
554                
555                if ( totalPaddingSize <= 0 ) {
556                    return toPad;
557                }
558            }
559                    
560            // get the lengths of padding for the left and right hand sides
561            
562            int leftPaddingSize  = 0;
563            int rightPaddingSize = 0;
564            
565            boolean padLeft  = (side & LEFT) > 0;
566            boolean padRight = (side & RIGHT) > 0;
567    
568            if ( padLeft && padRight ){
569                leftPaddingSize = totalPaddingSize / 2;
570                rightPaddingSize = totalPaddingSize - leftPaddingSize;
571            } else if ( padLeft ) {
572                leftPaddingSize = totalPaddingSize;
573            } else if ( padRight ) {
574                rightPaddingSize = totalPaddingSize;
575            }
576            
577            // build a string consisting only of padding with length size / 2 + 1
578            
579            int numRepeats = (size/2) + 1;
580            String repeatedPadding = repeat(padding, numRepeats);
581            
582            // build the final string
583            
584            StringBuffer buffer = new StringBuffer();
585            buffer.append( repeatedPadding.substring(0, leftPaddingSize) );
586            if ( toPad != null ) {
587                buffer.append( toPad );
588            }
589            buffer.append( repeatedPadding.substring(0, rightPaddingSize) );
590            
591            return buffer.toString();
592        }

594        /**
595         * Repeat a string <code>num</code> times
596         *
597         * <p>
598         * e.g. The following will produce the string <code>"hellohello"</code>
599         * </p>
600         * 
601         * <p>
602         * <code>
603         * String trimmed = StringUtils.repeat("hello,", 2);
604         * </code>
605         * </p>
606         *
607         * @param toRepeat  The string to repeat
608         * @param num       The number of times to repeat the string
609         *
610         * @return          The repeated string
611         *
612         * @since 0.1
613         */
614        public static String repeat(String toRepeat, int num) {
615    
616            // special cases when null, empty or less than or = zero
617            
618            if (toRepeat == null ) {
619                return null;
620            } 
621            if (toRepeat.equals("") || num <= 0) {
622                return "";
623            }
624            
625            StringBuffer buffer = new StringBuffer();
626            for (int i=0 ; i<num ; i++) {
627                buffer.append(toRepeat);
628            }
629            
630            return buffer.toString();
631        }
