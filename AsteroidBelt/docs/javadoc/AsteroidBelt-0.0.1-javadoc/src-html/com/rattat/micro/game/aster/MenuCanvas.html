013    package com.rattat.micro.game.aster;
014    
015    import javax.microedition.lcdui.Canvas;
016    import javax.microedition.lcdui.Font;
017    import javax.microedition.lcdui.Graphics;
018    
019    import com.rattat.micro.ui.selectslider.Option;
020    import com.rattat.micro.ui.selectslider.SelectListener;
021    import com.rattat.micro.ui.selectslider.SelectSlider;
022    
023    /**
024     * Canvas object for displaying options - new game, 
025     * turn sound off etc
026     *
027     * @author william@rattat.com
028     */
029    public class MenuCanvas extends Canvas implements SelectListener {
030    
031            /**
032             * The slider object which handles the option slider logic
033             */
034            private SelectSlider slider = null;
035        
036            /**
037             * The high score to display
038             */
039        private int highScore = 0;
040        
041        /**
042         * Title to display
043         */
044        private String title = "Asteroid Belt";
045        
046        /**
047         * Author to display
048         */
049        private String author = "by William Robertson";
050        
051        /**
052         * High score string
053         */
054        private String highScoreDescription = "high score: ";
055        
056        /**
057         * Info string
058         */
059        private String info = "www.rattat.com";
060        
061        /**
062         * The height of the rendered slider in pixels
063         */
064        private static final int SLIDER_HEIGHT = 50;
065        
066        /**
067         * Flag indicating if the menu should display the options
068         * available when paused or not (continue game)
069         */
070        private boolean paused = false;
071        
072        /**
073         * Flag indication whether sound is on or off - user will
074         * be given the option to turn it off/on accordingly
075         */
076        private boolean soundOn = false;
077        
078        /**
079         * Flag indication whether vibrations on or off - user will
080         * be given the option to turn them off/on accordingly
081         */
082        private boolean vibrateOn = false;
083        
084        /**
085         * Option to continue playing a paused game
086         */
087        public static final Option CONTINUE    = new Option(1, "continue game");
088        
089        /**
090         * Option to exit the application
091         */
092        public static final Option EXIT        = new Option(2, "exit");
093        
094        /**
095         * Option to turn sound on
096         */
097        public static final Option SOUND_ON    = new Option(3, "sound off");
098        
099        /**
100         * Option to turn sound off
101         */
102        public static final Option SOUND_OFF   = new Option(4, "sound on");
103        
104        /**
105         * Option to turn vibrations on
106         */
107        public static final Option VIBRATE_ON  = new Option(5, "vibrate off");
108        
109        /**
110         * Option to turn vibrations off
111         */
112        public static final Option VIBRATE_OFF = new Option(6, "vibrate on");
113        
114        /**
115         * Option to start new game
116         */
117        public static final Option NEW_GAME    = new Option(7, "new game");
118        
119        /**
120         * Option to display about page
121         */
122        public static final Option ABOUT       = new Option(8, "about");
123        
124        /**
125         * When left or right key is held down, slide then wait 
126         * for this number of miliseconds before sliding again
127         */
128        private static final int MOVE_WAIT_PERIOD = 500;
129        
130        /**
131         * Used to keep track of the last slide to stop very fast
132         * movement of the options
133         */
134        private long lastMoveTime = 0;
135    
136        /**
137         * Construct a new instance of MenuCanvas
138         * 
139         * @param slider
140         */
141        public MenuCanvas( SelectSlider slider ) {
142            this.slider = slider;
143            setFullScreenMode(true);
144            initSlider();
145        }
146        
147        /**
148         * Initialise the slider object
149         */
150        private void initSlider() {
151            slider.setLooped(true);
152            
153            slider.addOption(NEW_GAME);
154            slider.addOption(SOUND_ON);
155            slider.addOption(VIBRATE_ON);
156    
157            slider.addOption(ABOUT);
158            slider.addOption(EXIT);
159            slider.addSelectListener(this);
160            
161            slider.setCurrentOption(NEW_GAME);
162        }
163        
164        /**
165         * Canvas method
166         * 
167         * Listen for keypresses and move the move the slider appropriately
168         *
169         * @param keyCode
170         */
171        public void keyPressed(int keyCode) {
172            if ( System.currentTimeMillis() - lastMoveTime < MOVE_WAIT_PERIOD ) {
173                    return;
174            }
175    
176            int gameAction = getGameAction(keyCode);
177            boolean moved = false;
178            
179            if (gameAction == Canvas.LEFT) { 
180                slider.moveLeft();
181                moved = true;
182            } else if (gameAction == Canvas.RIGHT) {
183                slider.moveRight();
184                moved = true;
185            } else if (gameAction == Canvas.FIRE) {
186                slider.select();
187                moved = true;
188            }
189            
190            if ( moved ) {
191                    lastMoveTime = System.currentTimeMillis();
192                    repaint();
193            } 
194        }
195        
196        /**
197         * Canvas method - Key is released
198         */
199        public void keyReleased(int keyCode) {
200            int gameAction = getGameAction(keyCode);
201            
202            if (gameAction == Canvas.LEFT || gameAction == Canvas.RIGHT || gameAction == Canvas.FIRE) {
203                    lastMoveTime = 0;
204            }
205        }
206        
207        /**
208         * An option is selected - update options to reflect the change, if
209         * required
210         */
211        public void optionSelected(Option option) {
212            if (option == SOUND_ON) {
213                    setSoundOn(true);
214            } else if (option == SOUND_OFF) {
215                setSoundOn(false);
216            } else if (option == VIBRATE_ON) {
217                    setVibrateOn(true);
218            } else if (option == VIBRATE_OFF) {
219                setVibrateOn(false);
220            }
221        }
222        
223        /**
224         * Render the slider and information to the screen
225         */
226        public void paint(Graphics g) {
227            
228            // black background
229                    
230            g.setColor(0); // black
231            g.fillRect(0, 0, getWidth(), getHeight());
232            
233            // Draw the text
234            
235            int sliderTop = (getHeight() - SLIDER_HEIGHT) / 2;
236            int sliderBottom = (getHeight() + SLIDER_HEIGHT) / 2;
237            int anchor = Graphics.TOP|Graphics.HCENTER;
238            Font font = g.getFont();
239            int fontHeight = font.getHeight();
240            int yPos = 0;
241            int xPos = getWidth() / 2;
242            g.setColor(255, 255, 255); // white
243    
244            yPos = sliderTop - 3*fontHeight;
245            g.drawString(title, xPos, yPos, anchor);
246                    
247            yPos = sliderTop + -2*fontHeight;
248            g.drawString(author, xPos, yPos, anchor);
249            
250            if (highScore>0) {
251                yPos = sliderBottom + fontHeight;
252                g.drawString(highScoreDescription + highScore, xPos, yPos, anchor);
253            }
254            
255            yPos = sliderBottom + 2*fontHeight;
256            g.drawString(info, xPos, yPos, anchor);
257            
258            // draw the slider
259            
260            g.clipRect(0, 0, getWidth(), getHeight());
261            g.clipRect(0, sliderTop, getWidth(), SLIDER_HEIGHT);
262            
263            slider.paint(g);
264        }
265    
266        /**
267         * Alter the options to reflect a pause menu / 
268         * pre-game menu. Basically, hide the "continue game"
269         * option if unpaused
270         *
271         * @param paused
272         */
273            public void setPaused(boolean paused) {
274                    if ( paused && ! this.paused ) {        // if pausing unpaused menu
275                            slider.addOption(CONTINUE);
276                            slider.setCurrentOption(CONTINUE);
277                    }
278                    if ( ! paused && this.paused ) {        // if unpausing a paused menu
279                            slider.removeOption(CONTINUE);
280                            slider.setCurrentOption(NEW_GAME);
281                    }
282                    
283                    this.paused = paused;
284            }
285    
286        /**
287         * Alter the options to give hte user the option to 
288         * turn sound on / off according to {@code soundOn}
289         *
290         * @param soundOn
291         */
292            public void setSoundOn(boolean soundOn) {
293                    if ( soundOn && ! this.soundOn ) {      // if turning soundManager on when switched off
294                            slider.replaceOption(SOUND_ON, SOUND_OFF);
295                    }
296                    if ( ! soundOn && this.soundOn ) {      // if turning soundManager off when switched on
297                            slider.replaceOption(SOUND_OFF, SOUND_ON);
298                    }
299                    
300                    this.soundOn = soundOn;
301            }
302    
303        /**
304         * Alter the options to give the user the option to 
305         * turn vibrations on / off according to {@code vibrateOn}
306         *
307         * @param vibrateOn
308         */
309            public void setVibrateOn(boolean vibrateOn) {
310                    if ( vibrateOn && ! this.vibrateOn ) {  // if turning vibrate on when switched off
311                            slider.replaceOption(VIBRATE_ON, VIBRATE_OFF);
312                    }
313                    if ( ! vibrateOn && this.vibrateOn ) {  // if turning vibrate off when switched on
314                            slider.replaceOption(VIBRATE_OFF, VIBRATE_ON);
315                    }
316                    
317                    this.vibrateOn = vibrateOn;
318            }
319            
320            /**
321             * Set the high score that is displayed
322             * 
323             * @param highScore
324             */
325            public void setHighScore(int highScore) {
326                    this.highScore = highScore;
327            }
328    
329    }
