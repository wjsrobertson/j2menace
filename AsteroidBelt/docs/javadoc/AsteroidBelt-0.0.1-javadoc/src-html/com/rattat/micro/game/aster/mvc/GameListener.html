013    package com.rattat.micro.game.aster.mvc;
014    
015    /**
016     * <p>
017     * Defines an interface which objects wishing to observe
018     * game events must implement. 
019     * </p>
020     * 
021     * <p>
022     * Events are defined using the public static integers 
023     * like {@code EVENT_THRUST_START} and {@code EVENT_MISSILE_FIRED}
024     * which are defined in this interface.
025     * </p>
026     *
027     * @author william@rattat.com
028     */
029    public interface GameListener {
030            
031            /**
032             * Event fired when the game has first started
033             */
034        public static final int EVENT_GAME_START = 1;
035        
036            /**
037             * Event fired when the game ends
038             */
039        public static final int EVENT_GAME_END = 2;
040        
041            /**
042             * Event fired when an asteroid is destroyed
043             */
044        public static final int EVENT_ASTEROID_DESTROYED = 3;
045        
046            /**
047             * Event fired when the spaceship is destroyed
048             */
049        public static final int EVENT_SHIP_DESTROYED = 4;
050        
051            /**
052             * Event fired when a flying saucer is destroyed
053             */
054        public static final int EVENT_SAUCER_DESTROYED = 5;
055        
056        //public static final int EVENT_SHIP_MISSILE_FIRED = 6;
057        
058            /**
059             * Event fired when a flying saucer appears onscreen
060             */
061        public static final int EVENT_SAUCER_APPEAR = 7;
062        
063            /**
064             * Event fired when a flying saucer shoots a missile
065             */
066        public static final int EVENT_SAUCER_MISSILE_FIRED = 8;
067        
068            /**
069             * Event fired when a new level is entered
070             */
071        public static final int EVENT_NEW_LEVEL = 9;
072        
073            /**
074             * Event fired when a missilfew hundred milisecondse fired by a flying saucer 
075             * is blocked by an asteroid
076             */
077        public static final int EVENT_MISSILE_BLOCKED = 10;
078        
079            /**
080             * Event fired when the users space ship becomes mortal. This
081             * is a short time after gameplay starts to give the user a 
082             * chance to get into position when the l;evel first starts.
083             */
084        public static final int EVENT_SHIP_MORTAL = 11;
085                    
086        /**
087         * <p>
088         * Objects who wish to observe game events fired by the GameController
089         * class must implement this interface 
090         * </p>
091         * 
092         * <p>
093         * Objects can see and alter the state of the game by 
094         * performing operations on the {@code model} parameter.
095         * </p>
096         * 
097         * @param gameEvent The event that has been fired
098         * @param model             The model of the game
099         * 
100         * @see GameController.addListener(GameListener listener)
101         */
102            public void gameEvent(int gameEvent, Model model);
103    }
