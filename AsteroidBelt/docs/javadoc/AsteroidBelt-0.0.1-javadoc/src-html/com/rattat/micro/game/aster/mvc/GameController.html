029    /**
030     * <p>
031     * This class contains most of the actual logic that controls 
032     * and updates the model.
033     * </p>
034     * 
035     * <p>
036     * The {@code tick()} method is the method which is called externally
037     * to update the game state, incrementing elapsed time by one game
038     * tick.
039     * </p>
040     * 
041     * <p>
042     * This class broadcasts events which can be observed
043     * by clients who implement the {@code GameListener} interface
044     * after they have been added with {@code addListener()}.
045     * </p>
046     * 
047     * @author william@rattat.com
048     */

181        /**
182         * Create a set of random stars scattered around the background
183         * of the gameplay area
184         */
185        private void initStars() {
186            
187            Star[] stars = new Star[20];
188            
189            for (int i=0 ; i<20 ; i++) {
190                double width = 3 * random.nextDouble();
191                double x = gameScale * gameWidth * random.nextDouble();
192                double y = gameScale * gameHeight * random.nextDouble();
193                stars[i] = new Star(new Vector2D(x, y), width);
194            }
195            
196            model.setStars( stars );
197        }

200         * Initialise the space ship, reset it and
201         * place it in the starting position 
202         */
203        private void initSpaceShip() {
204            SpaceShip ship = model.getSpaceShip();
205            
206            ship.reset();
207            ship.setActive(true);
208            ship.setPosition(gameMaxX/2 * gameScale, gameMaxY/2 * gameScale);
209        }

211        /**
212         * <p>
213         * Create {@code numAsteroids} asteroids with random 
214         * location, angular velocity and velocity. 
215         * </p>
216         * 
217         * <p>
218         * This class uses the object pooling capabilities of the 
219         * {@code Model} to re-use inactive asteroids if they
220         * are available.
221         * </p>
222         * 
223         * @param numAsteroids      Num asteroids to create
224         * 
225         * @todo Improve random speed / position
226         */
227        private void initAsteroids(int numAsteroids) {
228    
229            for (int i=0 ; i<numAsteroids ; i++) {
230                    double xV = 2 * random.nextDouble() + 1;
231                    if ( random.nextInt(1) == 0 ) {
232                            xV = -xV;
233                    }
234                    double yV = 2 * random.nextDouble() + 1;
235                    if ( random.nextInt(1) == 0 ) {
236                            yV = -yV;
237                    }
238                    double xP = random.nextInt( (gameMaxX - gameMinX) * gameScale  );       // todo * scale
239                    double yP = random.nextInt( (gameMaxY - gameMinY) * gameScale );        // todo * scale
240                    
241                    Asteroid asteroid = model.nextFreeAsteroid();
242                    asteroid.set( new Vector2D(xP, yP), new Vector2D(xV, yV), 3 );
243            }
244        }

246        /**
247         * Place the flying saucer into gameplay at a random position 
248         * on either side of the screen
249         * 
250         * <p>
251             * Fires a {@code GameListener.EVENT_SAUCER_APPEAR} event 
252         * </p>
253         */
254        private void initSaucer() {
255            FlyingSaucer saucer = model.getSaucer(); 
256            
257            int type = random.nextInt(2);
258            int yPosition = (int) ((gameMaxY-gameMinY) * gameScale * random.nextDouble());
259                  
260            
261            saucer.setType(type);
262            saucer.reset();
263            saucer.updateForces();
264            saucer.setActive(true);
265            
266            if (type==FlyingSaucer.LEFT) {
267                saucer.setPosition(1, yPosition );
268            } else if (type==FlyingSaucer.RIGHT) {
269                saucer.setPosition(gameMaxX * gameScale -1, yPosition );
270            }
271            
272            notifyListeners( GameListener.EVENT_SAUCER_APPEAR );
273        }

275        /**
276         * <p>
277         * Update the game - increase by one unit of time
278         * </p>
279         */
280        public void tick() {
281            updateFloatingObjects();
282    
283            collisionCheck();
284    
285            nextLevelCheck();
286            
287            saucerBehaviour();
288            
289            gameOverCheck();
290        }

292        /**
293         * <p>
294         * Preform the logic required during gameplay
295         * to update the positions and inactive / active
296         * status of all {@code FloatingObject} objects 
297         * that are in play
298         * </p>
299         * 
300         * <p>
301             * Fires a {@code GameListener.EVENT_SHIP_MORTAL} event 
302             * if the spaceship has just reached the age where it 
303             * can be killed
304         * </p>
305         */
306        private void updateFloatingObjects() {
307    
308            // missile
309            
310            Missile missile = model.getMissile();
311            
312            if ( FloatingObject.isActive(missile) ) {
313                missile.update();  
314                correctPosition(missile);
315                
316                if (missile.getAge() > MISSLE_MAX_AGE_SHIP) {
317                    missile.setActive(false);
318                }
319            }
320            
321            // saucer missile
322            
323            Missile saucerMissile = model.getSaucerMissile();
324            
325            if ( FloatingObject.isActive( saucerMissile ) ) {
326                saucerMissile.update();  
327    
328                if (saucerMissile.getAge() > MISSLE_MAX_AGE_SAUCER) {
329                    saucerMissile.setActive(false);
330                }
331            }
332    
333            // asteroids
334            
335            Enumeration e = null;
336            Vector asteroids = model.getAsteroids();
337            
338            for ( e = asteroids.elements() ; e.hasMoreElements() ; ) {
339                Asteroid asteroid = (Asteroid) e.nextElement();
340                asteroid.update();  
341                asteroid.rotate();
342                correctPosition(asteroid);
343            }
344    
345            // spaceship
346            
347            SpaceShip spaceShip = model.getSpaceShip();
348    
349            if ( FloatingObject.isActive( spaceShip) ) {
350                spaceShip.update();
351                correctPosition(spaceShip);
352                if ( spaceShip.getAge() == SpaceShip.MORTAL_AGE ) {
353                    notifyListeners(GameListener.EVENT_SHIP_MORTAL);
354                }
355            } else {
356                model.incCreateShipTimer();
357                if (model.getCreateShipTimer() > CREATE_SHIP_WAIT && model.getLives() > 0) {
358                    initSpaceShip();
359                }
360            }
361            
362            // explosions
363            
364            Vector explosions = model.getExplosions();
365    
366            for (e = explosions.elements() ; e.hasMoreElements() ; ) {
367                ExplodingPolygon explosion = (ExplodingPolygon) e.nextElement();
368                explosion.update();  
369                if (explosion.getAge() > ExplodingPolygon.MAX_AGE) {
370                    explosion.setActive(false);
371                }
372                correctPosition(explosion);
373            }
374            
375            FlyingSaucer saucer = model.getSaucer();
376    
377            // flying saucer
378            
379            if ( FloatingObject.isActive(saucer) ) {
380                // update saucer position
381                saucer.update();
382    
383                if ( saucer.getPosition().getX() / gameScale > gameMaxX ) {
384                    saucer.setActive(false);
385                } else if ( saucer.getPosition().getX() / gameScale < gameMinX ) {
386                    saucer.setActive(false);
387                }
388    
389                correctPosition(saucer);
390            }
391        }

393        /**
394         * <p>
395         * Check for collisions between 
396         * astroids / missiles / space ship and the flying saucer
397         * </p>
398         * 
399         * <p>
400         * If a collision is detected then call the appropriate 
401         * methods to kiuck off explosions etc.
402         * </p>
403         * 
404         * <p>
405             * Fires a {@code GameListener.EVENT_MISSILE_BLOCKED} event 
406             * if a flying saucer missile gets blocked by an asteroid
407         * </p>
408         */
409        private void collisionCheck() {
410    
411            Vector asteroids      = model.getAsteroids();
412            SpaceShip spaceShip   = model.getSpaceShip();
413            Missile missile       = model.getMissile();
414            Missile saucerMissile = model.getSaucerMissile();
415            FlyingSaucer saucer   = model.getSaucer();
416            
417            // spaceship / asteroid intersection
418            
419            Enumeration e = null;
420    
421            if ( FloatingObject.isActive( spaceShip ) ) {
422                for ( e = asteroids.elements() ; e.hasMoreElements() ; ) {
423                    Asteroid asteroid = (Asteroid) e.nextElement();
424                    if ( FloatingObject.isActive( asteroid ) && spaceShip.isColliding( asteroid ) ) {
425                        shipDeath();
426                        break;
427                    }
428                }
429            }
430    
431            // spaceship / missile intersection
432    
433            if ( FloatingObject.isActive( spaceShip ) ) {
434                if ( FloatingObject.isActive( saucerMissile ) ) {
435                    if (spaceShip.isMortal() && spaceShip.isColliding(saucerMissile)) {
436                        shipDeath();
437                    }
438                }
439            }
440       
441            // missile / asteroid intersection
442            
443            for ( e = asteroids.elements(); e.hasMoreElements() ; ) {
444                Asteroid asteroid = (Asteroid) e.nextElement();
445                if ( FloatingObject.isActive(missile) && FloatingObject.isActive(asteroid) && asteroid.isColliding(missile)) {
446                    model.incScore( ASTEROID_HIT_POINTS );
447                    asteroidDeath(asteroid);
448                    missile.setActive(false);
449                } 
450                
451                // saucer can't shoot through asteroids
452                
453                if ( FloatingObject.isActive(saucerMissile) && FloatingObject.isActive(asteroid) && asteroid.isColliding(saucerMissile)) {
454                    saucerMissile.setActive(false);
455                    notifyListeners(GameListener.EVENT_MISSILE_BLOCKED);
456                }
457            }
458            
459            // flying saucer / missile intersection
460            
461            if ( FloatingObject.isActive(saucer) ) {
462                if ( FloatingObject.isActive(missile) ) {
463                    if ( saucer.isColliding(missile) ) {
464                            saucerDeath();
465                        missile.setActive(false);
466                    }
467                }
468            }
469            
470            // spaceship / flying saucerintersection
471    
472            if ( FloatingObject.isActive(spaceShip) ) {
473                if ( FloatingObject.isActive(saucer) ) {
474                    if (spaceShip.isMortal() && spaceShip.isColliding(saucer)) {
475                        shipDeath();
476                    }
477                }
478            }       
479        }

481        /**
482         * <p>
483         * Check if the player has destroyed all asteroids and 
484         * reached the next level
485         * </p>
486         * 
487         * <p>
488             * Fires a {@code GameListener.EVENT_NEW_LEVEL} event 
489             * if the next level has been reached
490         * </p>
491         */
492        private void nextLevelCheck() {
493            
494            // check to see if all asteroids are inactive (shot)
495            
496            boolean allInactive = true;
497            for ( Enumeration en=model.getAsteroids().elements() ; en.hasMoreElements() ; ) {
498                    Asteroid asteroid = (Asteroid) en.nextElement();
499                    if ( FloatingObject.isActive(asteroid) ) {
500                            allInactive = false;
501                    }
502            }
503            
504            // initialise next level if required
505            
506            if ( allInactive && model.getLives()>0) {
507                    model.incCurrentLevel();
508                initAsteroids( model.getCurrentLevel() );
509                initSpaceShip();
510                
511                notifyListeners( GameListener.EVENT_NEW_LEVEL );
512            }
513        }

515        /**
516         * <p>
517         * Handle the destruction of the flying saucer. Called when
518         * the player has shot a missile at the flying saucer. Allocates
519         * points, inactivates the saucer etc
520         * </p>
521         * 
522         * <p>
523             * Fires a {@code GameListener.EVENT_SAUCER_DESTROYED} event 
524         * </p>
525         */
526        private void saucerDeath() {
527            FlyingSaucer saucer = model.getSaucer(); 
528            
529            // create the explosion
530            
531            ExplodingPolygon explosion = model.nextFreeExplosion();
532            explosion.set(saucer, saucer.getPolygon());
533    
534            saucer.setActive(false); // deactivate the saucer
535            saucer.reset();
536    
537            model.incScore( SAUCER_HIT_POINTS );    // score points
538            notifyListeners( GameListener.EVENT_SAUCER_DESTROYED ); // notify observers of event
539        }

541        /**
542         * Handle the destruction of the player's space ship. 
543         * Loose a life and display explosion
544         * 
545         * <p>
546             * Fires a {@code GameListener.EVENT_SHIP_DESTROYED} event 
547         * </p>
548         */
549        private void shipDeath() {
550            SpaceShip spaceShip   = model.getSpaceShip();
551            
552            if (spaceShip!=null) {
553                
554                // create the explosion
555    
556                ExplodingPolygon explosion = model.nextFreeExplosion();
557                explosion.set(spaceShip, spaceShip.getPolygon());
558                
559                model.setLives( model.getLives() -1 );      // loose a life
560                model.setCreateShipTimer(0);
561       
562                spaceShip.setActive(false); // deactivate the ship
563                            
564                notifyListeners( GameListener.EVENT_SHIP_DESTROYED ); // notify observers of event
565            }
566        }

568        /**
569         * Handle the destruction of an asteroid. Create baby
570         * asteroids if the asteroid is large enough, create 
571         * explosion and allocate points.
572         * 
573         * <p>
574             * Fires a {@code GameListener.EVENT_ASTEROID_DESTROYED} event 
575         * </p>
576         * 
577         * @param asteroid  Asteroid to die
578         */
579        private void asteroidDeath(Asteroid asteroid) {
580    
581            asteroid.setActive(false); // deactivate the asteroid
582            int size = asteroid.getsize();
583            
584            // create baby asteroids, if required
585            
586            if ( size >1 ) {
587    
588                    Vector2D positiveVelocity = VMath2D.sum(
589                            asteroid.getVelocity().scaleBy( 3 + random.nextDouble() * 5 ).add( 1 - random.nextDouble() * 2 , 1 - random.nextDouble() * 3), 
590                            new Vector2D(5, 5)
591                    );
592                    Vector2D negativeVelocity = new Vector2D( -positiveVelocity.getX(), -positiveVelocity.getY() );
593                    
594                    Asteroid child1 = model.nextFreeAsteroid();
595                    child1.set( asteroid.getPosition(), positiveVelocity, size - 1 );
596    
597                    Asteroid child2 = model.nextFreeAsteroid();
598                    child2.set( asteroid.getPosition(), negativeVelocity, size - 1 );
599            }
600            
601            // create the explosion
602            
603            ExplodingPolygon explosion = model.nextFreeExplosion();
604            explosion.set(asteroid, asteroid.getPolygon());
605                   
606            notifyListeners( GameListener.EVENT_ASTEROID_DESTROYED ); // notify observers of event
607        }

609        /**
610         * If a game object was wandered offscreen then warp it around to 
611         * the other side.
612         * 
613         * @param object
614         */
615        private void correctPosition(FloatingObject object) {
616            Vector2D position = object.getPosition();
617    
618            if ( position.getX() / gameScale > gameMaxX ) {
619                position.setX((double)gameMinX * (double)gameScale);
620            } else if ( position.getX() / gameScale < gameMinX ) {
621                position.setX((double)gameMaxX * (double)gameScale);
622            }
623    
624            if ( position.getY() / gameScale > gameMaxY ) {
625                position.setY((double)gameMinY * (double)gameScale);
626            } else if ( position.getY() / gameScale < gameMinY ) {
627                position.setY((double)gameMaxY * (double)gameScale);
628            }
629        }

631        /**
632         * <p>
633         * Initliase the flying saucer at random intervals and 
634         * fire missile from saucer at random intervals
635         * </p>
636         * 
637         * <p>
638             * Fires a {@code GameListener.EVENT_SAUCER_MISSILE_FIRED} event if
639             * a missile is fired 
640         * </p>
641         */
642        public void saucerBehaviour() {
643            
644            FlyingSaucer saucer   = model.getSaucer();
645            
646            if ( ! FloatingObject.isActive(saucer) && random.nextDouble() * SAUCER_ACTIVITY < 1 ) {      // todo 
647                initSaucer();
648            }
649            
650            if ( FloatingObject.isActive(saucer) ) {
651                    Missile saucerMissile = model.getSaucerMissile();
652                SpaceShip spaceShip   = model.getSpaceShip();
653                    
654                // fire a missile if the time is ripe
655                if ( ! FloatingObject.isActive(saucerMissile) 
656                            && FloatingObject.isActive(saucer) 
657                            && FloatingObject.isActive(spaceShip)
658                            && spaceShip.isMortal() 
659                            && saucer.getAge() > MIN_KILL_AGE
660                            && random.nextInt( SAUCER_MISSILE_ACTIVITY ) < 1
661                            
662                ) {
663    
664                    saucerMissile.reset();
665                    saucerMissile.setPosition(saucer.getPosition());
666                    Vector2D directionVector = VMath2D.difference(spaceShip.getPosition(), saucer.getPosition());
667                    saucerMissile.setVelocity( directionVector.normalize().scaleBy( saucerMissile.getMaxSpeed() ) );
668                    saucerMissile.setActive(true);
669                    saucerMissile.setAge(0);
670                    
671                    notifyListeners( GameListener.EVENT_SAUCER_MISSILE_FIRED );
672                }       
673            } 
674        }

676        /**
677         * <p>
678         * Setup the model and prepare to play a new game
679         * </p>
680         * 
681         * <p>
682             * Fire a {@code GameListener.EVENT_GAME_START} 
683         * </p>
684         */
685        public void newGame() {
686            model.setScore(0);
687            model.setLives( NUM_LIVES_START );
688            model.setCurrentLevel( 1 );
689            model.getAsteroids().removeAllElements();
690            model.getExplosions().removeAllElements();
691            model.getSaucerMissile().setActive(false);
692            model.getSaucer().setActive(false);
693            model.getSpaceShip().setActive(true);
694    
695            initAsteroids( model.getCurrentLevel() );
696            initSpaceShip();
697            initStars();
698    
699            notifyListeners( GameListener.EVENT_GAME_START );
700        }

702        /**
703         * <p>
704         * Check if all asteroids have been destroyed and the
705         * game is over. 
706         * </p>
707         * 
708         * <p>
709         * Fires a {@code GameListener.EVENT_GAME_END} 
710         * event is it is over.
711         * </p>
712         */
713        private void gameOverCheck() {
714            if (model.getLives() <= 0) {
715                model.incDeadTimer();
716                if (model.getDeadTimer() > DEAD_GAMEOVER_WAIT) {
717                    model.setDeadTimer(-1);
718                    notifyListeners( GameListener.EVENT_GAME_END );
719                }
720            }
721        }

723        /**
724         * Add a new listener to observe game events
725         * 
726         * @param listener
727         */
728        public void addListener(GameListener listener) {
729            if ( ! listeners.contains(listener) ) {
730                    listeners.addElement(listener);
731            }
732        }
733        
734        /**
735         * Remove a game listener 
736         * 
737         * @param listener
738         */
739        public void removeListener(GameListener listener) {
740            if ( listeners.contains(listener) ) {
741                    listeners.removeElement(listener);
742            }
743        }
744        
745        /**
746         * Notify game listeners of game events
747         * 
748         * @param listener
749         */
750        private void notifyListeners(int event) {
751            for ( Enumeration e = listeners.elements() ; e.hasMoreElements() ; ) {
752                    GameListener listener = (GameListener) e.nextElement();
753                    
754                    listener.gameEvent(event, model);
755            }
756        }
