334            /**
335             * <p>
336             * Get an inactive Asteroid to use or create a new
337             * one if there are no inactive ones.
338             * </p>
339             * 
340             * <p>
341             * The Asteroid will be activated before it is returned
342             * </p>
343             * 
344             * <p>
345             * This is a simple form of object pooling
346             * </p>
347             * 
348             * @return
349             */
350            public Asteroid nextFreeAsteroid() {
351                    Asteroid asteroid = null;
352                    
353                    // check if there is an inactive Asteroid to use
354                    
355                    for ( Enumeration en = asteroids.elements() ; en.hasMoreElements() ; ) {
356                            asteroid = (Asteroid) en.nextElement();
357                            if ( ! asteroid.isActive() ) {
358                                    asteroid.setActive(true);
359                                    return asteroid;
360                            }
361                    }
362                    
363                    // no existing asteroids found, so create a new one
364                    
365                    asteroid = new Asteroid();
366            asteroid.setActive(true);
367            if ( ! asteroids.contains(asteroid) ) {
368                    asteroids.addElement(asteroid);
369            }
370                    
371                    return asteroid;
372            }
373            
374            /**
375             * <p>
376             * Get an inactive ExplodingPolygon to use or create a new
377             * one if there are no inactive ones.
378             * </p>
379             * 
380             * <p>
381             * The ExplodingPolygon will be activated before it is returned
382             * </p>
383             * 
384             * <p>
385             * This is a simple form of object pooling
386             * </p>
387             * 
388             * @return
389             */
390            public ExplodingPolygon nextFreeExplosion() {
391                    ExplodingPolygon explosion = null;
392                    
393                    // check if there is an inactive ExplodingPolygon to use
394                    
395                    for ( Enumeration en = explosions.elements() ; en.hasMoreElements() ; ) {
396                            explosion = (ExplodingPolygon) en.nextElement();
397                            if ( ! explosion.isActive() ) {
398                                    explosion.reset();
399                                    explosion.setActive(true);
400                                    return explosion;
401                            }
402                    }
403                    
404                    // no existing explosions found, so create a new one
405                    
406                    explosion = new ExplodingPolygon();
407                    explosion.setActive(true);
408                    explosions.addElement(explosion);
409                    
410                    return explosion;
411            }
