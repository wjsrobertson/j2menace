011    package com.rattat.micro.ui.vibrator;
012    
013    import javax.microedition.lcdui.Display;
014    
015    /**
016     * Class for performing vibrations.
017     *
018     * Needs to have a Display object set with setDisplay or it won't
019     * do anything.
020     *
021     * All methods must be called statically
022     *
023     * @author poobar
024     */
025    public final class Vibrator {
026        
027        /**
028         * Private constructor to stop instantiation
029         */
030        private Vibrator() {
031        }
032        
033        /**
034         * The Display instance which is uysed for vibrations
035         */
036        private static Display display;
037        
038        /**
039         * Flag defining whether ot not the vibrator is active
040         */
041        private static boolean on = true;
042    
043        /**
044         * Set the display object, used to perform the vibration 
045         *
046         * @param Display display 
047         */
048        public static void setDisplay(Display display) {
049            Vibrator.display = display;
050        }
051        
052        /**
053         * Vibrate for a length of time
054         *
055         * Will quietly fail if there is no display object set
056         *
057         * @param duration  Duration in miliseconds
058         */
059        public static void vibrate(int duration) {
060            if (display != null && on) {
061                display.vibrate(duration);
062            }
063        }
064        
065        /**
066         * Check if vibrations are turned on or off
067         *
068         * @return 
069         */
070        public static boolean isOn() {
071            return on;
072        }
073        
074        /**
075         * Turn vibrations on or off
076         *
077         * @return 
078         */
079        public static void setOn(boolean on) {
080            Vibrator.on = on;
081        }
082        
083    }
