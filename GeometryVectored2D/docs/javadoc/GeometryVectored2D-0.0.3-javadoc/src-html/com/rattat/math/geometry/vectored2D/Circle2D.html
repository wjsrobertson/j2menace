010    package com.rattat.math.geometry.vectored2D;
011    
012    /**
013     *
014     * @author poobar
015     */
016    public class Circle2D {
017        
018        private double radius = 0;
019        
020        private Vector2D center = new Vector2D();
021        
022        private boolean modified = true;
023        
024        private BoundingRectangle2D bounds = null;
025        
026        /** Creates a new instance of Circle2D */
027        public Circle2D() {
028        }
029        
030        /** Creates a new instance of Circle2D */
031        public Circle2D(Vector2D center, double radius) {
032            setCenter( new Vector2D(center) );
033            setRadius(radius);
034        }
035    
036        public double getRadius() {
037            return radius;
038        }
039    
040        public void setRadius(double radius) {
041            this.radius = radius;
042        }
043    
044        public Vector2D getCenter() {
045            return center;
046        }
047    
048        public void setCenter(Vector2D center) {
049            this.center = center;
050        }
051        
052        public BoundingRectangle2D getBounds() {
053            if ( bounds == null ) {
054                bounds = new BoundingRectangle2D();
055            }
056            
057            double diameter = 2 * radius;
058            
059            bounds.set( center.getX()-radius, center.getY()-radius, diameter, diameter );
060            
061            return bounds;
062        }
063        
064        /**
065         * Check if a point is contained within this circle
066         * 
067         * @param point Point to check
068         *
069         * @return true if containted, false otherwise
070         */
071        public boolean contains(Vector2D point) {
072            if (! getBounds().contains(point)) {
073                return false;
074            }
075            
076            // x*x + y*y - R*R = 0 -> equation for radius
077            
078            double x2 = point.getX() * point.getX();
079            double y2 = point.getX() * point.getX();
080            double r2 = radius * radius;
081            
082            return (x2 + y2 - r2) < 0;
083        }
084    
085        /**
086         * Check if a line intersects this circle
087         * 
088         * @param line Line to check
089         *
090         * @return true if intersects, false otherwise
091         */
092        public boolean intersects(Line2D line) {
093            if (! line.getBounds().intersects(getBounds())) {
094                return false;
095            }
096    
097            return true;    
098        }
099    
100        /**
101         * Check if this triangle intersects another 
102         * 
103         * @param circle Circle to check
104         *
105         * @return true if intersects, false otherwise
106         */
107        public boolean intersects(Circle2D circle) {
108            if (! circle.getBounds().intersects(getBounds())) {
109                return false;
110            }
111            
112            return (radius + circle.getRadius()) > center.distance( circle.getCenter() );
113        }
114        
115    }
